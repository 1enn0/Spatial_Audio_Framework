29 (
30  void ** const phBeam
31 )
32 {
33  beamformer_data* pData = (beamformer_data*)malloc1d(sizeof(beamformer_data));
34  *phBeam = (void*)pData;
35  int i, ch;
36 
37  /* default user parameters */
38  pData->beamOrder = 1;
39  for(i=0; i<MAX_NUM_BEAMS; i++){
40  pData->beam_dirs_deg[i][0] = __default_LScoords64_rad[i][0]*180.0f/M_PI;
41  pData->beam_dirs_deg[i][1] = (__default_LScoords64_rad[i][1] - M_PI/2.0f) < -M_PI/2.0f ?
42  (M_PI/2.0f + __default_LScoords64_rad[i][1]) : (__default_LScoords64_rad[i][1] - M_PI/2.0f);
43  pData->beam_dirs_deg[i][1] *= 180.0f/M_PI;
44  }
45  pData->nBeams = 1;
46  pData->beamType = STATIC_BEAM_TYPE_HYPERCARDIOID;
47  pData->chOrdering = CH_ACN;
48  pData->norm = NORM_SN3D;
49 
50  /* flags */
51  for(ch=0; ch<MAX_NUM_BEAMS; ch++)
52  pData->recalc_beamWeights[ch] = 1;

93 void beamformer_process
94 (
95  void * const hBeam,
96  const float *const * inputs,
97  float ** const outputs,
98  int nInputs,
99  int nOutputs,
100  int nSamples
101 )
102 {
103  beamformer_data *pData = (beamformer_data*)(hBeam);
104  int ch, i, bi, nSH, mixWithPreviousFLAG;
105  float c_n[MAX_SH_ORDER+1];
106 
107  /* local copies of user parameters */
108  int nBeams, beamOrder;
109  NORM_TYPES norm;
110  CH_ORDER chOrdering;
111  beamOrder = pData->beamOrder;
112  nSH = ORDER2NSH(beamOrder);
113  nBeams = pData->nBeams;
114  norm = pData->norm;
115  chOrdering = pData->chOrdering;
116 
117  /* Apply beamformer */
118  if(nSamples == BEAMFORMER_FRAME_SIZE) {
119  /* Load time-domain data */
120  for(i=0; i < SAF_MIN(nSH, nInputs); i++)
121  utility_svvcopy(inputs[i], BEAMFORMER_FRAME_SIZE, pData->SHFrameTD[i]);
122  for(; i<MAX_NUM_SH_SIGNALS; i++)
123  memset(pData->SHFrameTD[i], 0, BEAMFORMER_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
124 
125  /* account for input channel order convention */
126  switch(chOrdering){
127  case CH_ACN: /* already ACN, do nothing*/ break; /* Otherwise, convert to ACN... */
128  case CH_FUMA: convertHOAChannelConvention((float*)pData->SHFrameTD, beamOrder, BEAMFORMER_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
129  }
130 
131  /* account for input normalisation scheme */
132  switch(norm){
133  case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
134  case NORM_SN3D: convertHOANormConvention((float*)pData->SHFrameTD, beamOrder, BEAMFORMER_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
135  case NORM_FUMA: convertHOANormConvention((float*)pData->SHFrameTD, beamOrder, BEAMFORMER_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
136  }
137 
138  /* Calculate beamforming coeffients */
139  mixWithPreviousFLAG = 0;
140  for(bi=0; bi<nBeams; bi++){
141  if(pData->recalc_beamWeights[bi]){
142  memset(pData->beamWeights[bi], 0, MAX_NUM_SH_SIGNALS*sizeof(float));
143  switch(pData->beamType){
144  case STATIC_BEAM_TYPE_CARDIOID: beamWeightsCardioid2Spherical(beamOrder, c_n); break;
145  case STATIC_BEAM_TYPE_HYPERCARDIOID: beamWeightsHypercardioid2Spherical(beamOrder, c_n); break;
146  case STATIC_BEAM_TYPE_MAX_EV: beamWeightsMaxEV(beamOrder, c_n); break;
147  }
148  rotateAxisCoeffsReal(beamOrder, (float*)c_n, SAF_PI/2.0f - pData->beam_dirs_deg[bi][1]*SAF_PI/180.0f,
149  pData->beam_dirs_deg[bi][0]*SAF_PI/180.0f, (float*)pData->beamWeights[bi]);
150  pData->recalc_beamWeights[bi] = 0;
151  mixWithPreviousFLAG = 1;
152  }
153  }
154 
155  /* Apply beam weights */
156  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nBeams, BEAMFORMER_FRAME_SIZE, nSH, 1.0f,
157  (const float*)pData->beamWeights, MAX_NUM_SH_SIGNALS,
158  (const float*)pData->prev_SHFrameTD, BEAMFORMER_FRAME_SIZE, 0.0f,
159  (float*)pData->outputFrameTD, BEAMFORMER_FRAME_SIZE);
160 
161  /* Fade between (linearly inerpolate) the new weights and the previous weights (only if the new weights are different) */
162  if(mixWithPreviousFLAG){
163  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nBeams, BEAMFORMER_FRAME_SIZE, nSH, 1.0f,
164  (float*)pData->prev_beamWeights, MAX_NUM_SH_SIGNALS,
165  (float*)pData->prev_SHFrameTD, BEAMFORMER_FRAME_SIZE, 0.0f,
166  (float*)pData->tempFrame, BEAMFORMER_FRAME_SIZE);
167 
168  /* Apply the linear interpolation */
169  for (i=0; i < nBeams; i++){
170  utility_svvmul((float*)pData->interpolator_fadeIn, (float*)pData->outputFrameTD[i], BEAMFORMER_FRAME_SIZE, (float*)pData->outputFrameTD_fadeIn[i]);
171  utility_svvmul((float*)pData->interpolator_fadeOut, (float*)pData->tempFrame[i], BEAMFORMER_FRAME_SIZE, (float*)pData->tempFrame_fadeOut[i]);
172  }
173  cblas_scopy(nBeams*BEAMFORMER_FRAME_SIZE, (float*)pData->outputFrameTD_fadeIn, 1, (float*)pData->outputFrameTD, 1);
174  cblas_saxpy(nBeams*BEAMFORMER_FRAME_SIZE, 1.0f, (float*)pData->tempFrame_fadeOut, 1, (float*)pData->outputFrameTD, 1);
175 
176  /* for next frame */
177  utility_svvcopy((const float*)pData->beamWeights, MAX_NUM_BEAMS*MAX_NUM_SH_SIGNALS, (float*)pData->prev_beamWeights);
178  }
179 
180  /* for next frame */
181  utility_svvcopy((const float*)pData->SHFrameTD, MAX_NUM_SH_SIGNALS*BEAMFORMER_FRAME_SIZE, (float*)pData->prev_SHFrameTD);
182 
183  /* copy to output buffer */
184  for(ch = 0; ch < SAF_MIN(nBeams, nOutputs); ch++)
185  utility_svvcopy(pData->outputFrameTD[ch], BEAMFORMER_FRAME_SIZE, outputs[ch]);
186  for (; ch < nOutputs; ch++)
187  memset(outputs[ch], 0, BEAMFORMER_FRAME_SIZE*sizeof(float));
188  }
189  else
190  for (ch=0; ch < nOutputs; ch++)
191  memset(outputs[ch], 0, BEAMFORMER_FRAME_SIZE*sizeof(float));
192 }
