42 # endif

44 # define SAF_SLEEP(msecs) do { \
45  struct timespec ts; \
46  ts.tv_sec = msecs/1000; \
47  ts.tv_nsec = msecs%1000*1000; \
48  nanosleep(&ts, NULL); \
49 } while (0)
