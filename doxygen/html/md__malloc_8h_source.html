22 /*
23  * Filename:
24  * md_malloc.h
25  * Description:
26  * Implementations of dynamic memory allocation functions for
27  * contiguous multidimensional "arrays".
28  * Taken from: https://github.com/leomccormack/md_malloc
29  * Example:
30  * An example of allocating, indexing and freeing a 3-D "array":
31  * float*** example3D = (float***)malloc3d(10, 20, 5, sizeof(float);
32  * // Due to the contiguous nature of the allocation, this is possible:
33  * memset(&example3D[0][0][0], 0, 10*20*5*sizeof(float));
34  * // And my still be indexed normally as:
35  * example3D[3][19][2] = 22.0f;
36  * // To free, simply call:
37  * free(example3D);
38  * // Or:
39  * free3d((void***)example3D);
40  * Dependencies:
41  * (none)
42  * Author, date created:
43  * Leo McCormack, 11.06.2019
44  */
45

53 /* These macros return a pointer to the address of the first element in the
54  * array. Use them for passing arrays to memset/memcpy, or blas/lapack functions
55  * etc. e.g.
56  * float** array2D = (float**)malloc2d(10, 40, sizeof(float));
57  * memset(ADR2D(array2D), 0, 10*40*sizeof(float)); */
58 #define ADR1D(A) (&A[0])
59 #define ADR2D(A) (&A[0][0])
60 #define ADR3D(A) (&A[0][0][0])
