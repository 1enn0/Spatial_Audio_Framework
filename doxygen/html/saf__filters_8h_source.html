109 /*
110  * Function: getOctaveBandCutoffFreqs
111  * ----------------------------------
112  * Converts octave band CENTRE frequencies into CUTOFF frequencies.
113  * Note: the lower and upper CENTRE frequencies only have their upper and lower
114  * CUTOFF frequencies computed, respectively. e.g.:
115  * centreFreqs[6] = { 125, 250, 500, 1000, 2000, 4000 }, becomes:
116  * cutoffFreqs[5] = { 176, 354, 707, 1410, 2830 }
117  * Passing cutoffFreqs[5] to "FIRFilterbank", will give filter coefficients for
118  * the following:
119  * Band1: LPF @ 176Hz
120  * Band2: BFP @ 176-354Hz
121  * Band3: BFP @ 354-707Hz
122  * Band4: BFP @ 707-1410Hz
123  * Band5: BFP @ 1410-2830Hz
124  * Band6: HPF @ 2830Hz
125  * (Basically, band 125Hz also encapsulates everything down to DC, and band 4kHz
126  * also encapsulates everything up to Nyquist)
127  * Also note: cutoffFreqs vector is shorter than centreFreqs by 1 element.
128  *
129  * Input Arguments:
130  * centreFreqs - centre frequencies (octave bands); nCentreFreqs x 1
131  * nCentreFreqs - number of centre frequencies
132  * Output Arguments:
133  * cutoffFreqs - cutoff frequencies, which encapsulate the specified centre
134  * frequencies by 1 octave; (nCentreFreqs-1) x 1
135  */
136 void getOctaveBandCutoffFreqs(float* centreFreqs,
137  int nCentreFreqs,
138  float* cutoffFreqs);

241 /*
242  * Function: FIRCoeffs
243  * -------------------
244  * FIR filters by windowing. When using the Hamming window, and scalingFLAG=1,
245  * the function is numerically identical to the default 'fir1' function in
246  * Matlab (when using it in single precision mode) [1].
247  * Note: input argument 'order' cannot be odd valued.
248  *
249  * Some guidelines regarding the approx order (N) for certain filters. i.e.
250  * the orders where you actually get the expected -6dB attenuation at the cutoff
251  * frequency specified (fs=48kHz, Hamming window, scalingFLAG=1).
252  * (Use these figures only just to get a rough idea)
253  * - LPF @ 100Hz - N~1400
254  * - LPF @ 250Hz - N~550
255  * - LPF @ 1kHz - N~150
256  * - LPF @ 4kHz - N~40
257  * - BPF @ 88-176Hz - N~2500
258  * - BPF @ 176-354Hz - N~1600
259  * - BPF @ 707-1410Hz - N~400
260  * - HPF @ 200Hz - N~450
261  * - HPF @ 4kHz - N~60
262  *
263  * Input Arguments:
264  * filterType - see 'FIR_FILTER_TYPES' enum
265  * order - filter order (N). Must be even.
266  * cutoff1 - filter1 cutoff in Hz, for LPF/HPF, and lower cutoff for
267  * BPF/BSF
268  * cutoff2 - filter2 cutoff in Hz, not needed for LPF/HPF, this is the
269  * upper cutoff for BPF/BSF
270  * samplerate - sampling rate in Hz
271  * windowType - see 'WINDOWING_FUNCTION_TYPES' enum
272  * scalingFLAG - 0: none, 1: scaling applied to ensure passband is at 0dB
273  * Output Arguments:
274  * filter - filter coefficients/weights/taps; (order+1) x 1
275  *
276  * [1] "Programs for Digital Signal Processing", IEEE Press John Wiley & Sons,
277  * 1979, pg. 5.2-1.
278  */
279 void FIRCoeffs(FIR_FILTER_TYPES filterType,
280  int order,
281  float cutoff1,
282  float cutoff2,
283  float sampleRate,
284  WINDOWING_FUNCTION_TYPES windowType,
285  int scalingFLAG,
286  float* filter);

288 /*
289  * Function: FIRFilterbank
290  * -----------------------
291  * Returns a bank of FIR filter coefficients required to divide a signal into
292  * frequency bands. Provided the order is sufficient, the sum of the bands
293  * should recontruct the original (although, shifted in time due to group delay)
294  * e.g fc[1] = { 1000 };
295  * Band1, &filter[0*(order+1)] : LPF @ 1kHz
296  * Band2, &filter[1*(order+1)] : HPF @ 1kHz
297  * e.g fc[3] = { 1000, 2000, 4000 };
298  * Band1, &filter[0*(order+1)] : LPF @ 1kHz
299  * Band2, &filter[1*(order+1)] : BPF @ 1-2kHz
300  * Band3, &filter[2*(order+1)] : BPF @ 2-4kHz
301  * Band4, &filter[3*(order+1)] : HPF @ 4kHz
302  *
303  * Input Arguments:
304  * order - filter order. Must be even.
305  * fc - vector of cutoff frequencies; nCutoffFreqs x 1
306  * nCutoffFreqs - number of cutoff frequencies in vector 'fc'.
307  * samplerate - sampling rate in Hz
308  * windowType - see 'WINDOWING_FUNCTION_TYPES' enum
309  * scalingFLAG - 0: none, 1: scaling applied to ensure passbands are at 0dB
310  * Output Arguments:
311  * filter - filter coefficients/weights/taps;
312  * FLAT: (nCutoffFreqs+1) x (order+1)
313  */
314 void FIRFilterbank(int order,
315  float* fc,
316  int nCutoffFreqs,
317  float sampleRate,
318  WINDOWING_FUNCTION_TYPES windowType,
319  int scalingFLAG,
320  float* filterbank);
