47 void ambi_dec_create
48 (
49  void ** const phAmbi
50 )
51 {

53  *phAmbi = (void*)pData;
54  int i, j, ch, band;
55 
56  /* default user parameters */
57  loadLoudspeakerArrayPreset(LOUDSPEAKER_ARRAY_PRESET_T_DESIGN_24, pData->loudpkrs_dirs_deg, &(pData->new_nLoudpkrs), &(pData->loudpkrs_nDims));

62  pData->enableHRIRsPreProc = 1;

65  pData->norm = NORM_SN3D;

69  pData->rE_WEIGHT[1] = 1;
70  pData->diffEQmode[0] = ENERGY_PRESERVING;
71  pData->diffEQmode[1] = ENERGY_PRESERVING;

179 void ambi_dec_initCodec
180 (
181  void* const hAmbi
182 )
183 {
184  ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
185  ambi_dec_codecPars* pars = pData->pars;
186  int i, ch, d, j, n, ng, nGrid_dirs, masterOrder, nSH_order, max_nSH, nLoudspeakers;
187  float* grid_dirs_deg, *Y, *M_dec_tmp, *g, *a, *e, *a_n, *hrtf_vbap_gtable;;
188  float a_avg[MAX_SH_ORDER], e_avg[MAX_SH_ORDER], azi_incl[2], sum_elev;
189 #ifdef SAF_ENABLE_SOFA_READER_MODULE
190  SAF_SOFA_ERROR_CODES error;
191  saf_sofa_container sofa;
192 #endif
193 
194  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
195  return; /* re-init not required, or already happening */
196  while (pData->procStatus == PROC_STATUS_ONGOING){
197  /* re-init required, but we need to wait for the current processing loop to end */
198  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
199  SAF_SLEEP(10);
200  }
201 
202  /* for progress bar */
203  pData->codecStatus = CODEC_STATUS_INITIALISING;
204  strcpy(pData->progressBarText,"Initialising");
205  pData->progressBar0_1 = 0.0f;
206 
207  /* reinit afSTFT */
208  masterOrder = pData->new_masterOrder;
209  max_nSH = (masterOrder+1)*(masterOrder+1);
210  nLoudspeakers = pData->new_nLoudpkrs;
211  if(pData->hSTFT==NULL){
212  if(pData->new_binauraliseLS)
213  afSTFT_create(&(pData->hSTFT), max_nSH, NUM_EARS, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
214  else
215  afSTFT_create(&(pData->hSTFT), max_nSH, nLoudspeakers, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
216  afSTFT_clearBuffers(pData->hSTFT);
217  }
218  else{
219  if(pData->new_binauraliseLS)
220  afSTFT_channelChange(pData->hSTFT, max_nSH, NUM_EARS);
221  else
222  afSTFT_channelChange(pData->hSTFT, max_nSH, nLoudspeakers);
223  afSTFT_clearBuffers(pData->hSTFT);
224  }
225  pData->binauraliseLS = pData->new_binauraliseLS;
226  pData->nLoudpkrs = nLoudspeakers;
227 
228  /* Quick and dirty check to find loudspeaker dimensionality */
229  strcpy(pData->progressBarText,"Computing decoder");
230  pData->progressBar0_1 = 0.2f;
231  sum_elev = 0.0f;
232  for(ch=0; ch < nLoudspeakers; ch++)
233  sum_elev += fabsf(pData->loudpkrs_dirs_deg[ch][1]);
234  if( (((sum_elev < 5.0f) && (sum_elev > -5.0f))) || (nLoudspeakers < 4) )
235  pData->loudpkrs_nDims = 2;
236  else
237  pData->loudpkrs_nDims = 3;
238 
239  /* add virtual loudspeakers for 2D case */
240  if (pData->loudpkrs_nDims == 2){
241  assert(nLoudspeakers<=MAX_NUM_LOUDSPEAKERS-2);
242  pData->loudpkrs_dirs_deg[nLoudspeakers][0] = 0.0f;
243  pData->loudpkrs_dirs_deg[nLoudspeakers][1] = -90.0f;
244  pData->loudpkrs_dirs_deg[nLoudspeakers+1][0] = 0.0f;
245  pData->loudpkrs_dirs_deg[nLoudspeakers+1][1] = 90.0f;
246  nLoudspeakers += 2;
247  }
248 
249  /* prep */
250  nGrid_dirs = 480; /* Minimum t-design of degree 30, has 480 points */
251  g = malloc1d(nLoudspeakers*sizeof(float));
252  a = malloc1d(nGrid_dirs*sizeof(float));
253  e = malloc1d(nGrid_dirs*sizeof(float));
254 
255  /* calculate loudspeaker decoding matrices */
256  for( d=0; d<NUM_DECODERS; d++){
257  M_dec_tmp = malloc1d(nLoudspeakers * max_nSH * sizeof(float));
258  switch(pData->dec_method[d]){
259  case DECODING_METHOD_SAD:
260  getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_SAD, masterOrder, 0, M_dec_tmp);
261  break;
262  case DECODING_METHOD_MMD:
263  getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_MMD, masterOrder, 0, M_dec_tmp);
264  break;
265  case DECODING_METHOD_EPAD:
266  getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_EPAD, masterOrder, 0, M_dec_tmp);
267  break;
268  case DECODING_METHOD_ALLRAD:
269  getLoudspeakerDecoderMtx((float*)pData->loudpkrs_dirs_deg, nLoudspeakers, LOUDSPEAKER_DECODER_ALLRAD, masterOrder, 0, M_dec_tmp);
270  break;
271  }
272 
273  /* diffuse-field EQ for orders 1..masterOrder */
274  for( n=1; n<=masterOrder; n++){
275  /* truncate M_dec for each order */
276  nSH_order = (n+1)*(n+1);
277  free(pars->M_dec[d][n-1]);
278  pars->M_dec[d][n-1] = malloc1d(nLoudspeakers* nSH_order * sizeof(float));
279  free(pars->M_dec_cmplx[d][n-1]);
280  pars->M_dec_cmplx[d][n-1] = malloc1d(nLoudspeakers * nSH_order * sizeof(float_complex));
281  for(i=0; i<nLoudspeakers; i++){
282  for(j=0; j<nSH_order; j++){
283  pars->M_dec[d][n-1][i*nSH_order+j] = M_dec_tmp[i*max_nSH +j]; /* for applying in the time domain, and... */
284  pars->M_dec_cmplx[d][n-1][i*nSH_order+j] = cmplxf(pars->M_dec[d][n-1][i*nSH_order+j], 0.0f); /* for the time-frequency domain */
285  }
286  }
287 
288  /* create dedicated maxrE weighted versions */
289  a_n = malloc1d(nSH_order*nSH_order*sizeof(float));
290  getMaxREweights(n, 1, a_n); /* weights returned as diagonal matrix */
291  free(pars->M_dec_maxrE[d][n-1]);
292  pars->M_dec_maxrE[d][n-1] = malloc1d(nLoudspeakers * nSH_order * sizeof(float));
293  free(pars->M_dec_cmplx_maxrE[d][n-1]);
294  pars->M_dec_cmplx_maxrE[d][n-1] = malloc1d(nLoudspeakers * nSH_order * sizeof(float_complex));
295  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLoudspeakers, nSH_order, nSH_order, 1.0f,
296  pars->M_dec[d][n-1], nSH_order,
297  a_n, nSH_order, 0.0f,
298  pars->M_dec_maxrE[d][n-1], nSH_order); /* for applying in the time domain */
299  for(i=0; i<nLoudspeakers * nSH_order; i++)
300  pars->M_dec_cmplx_maxrE[d][n-1][i] = cmplxf(pars->M_dec_maxrE[d][n-1][i], 0.0f); /* for the time-frequency domain */
301 
302  /* fire a plane-wave from each grid direction to find the total energy/amplitude (using non-maxrE weighted versions) */
303  Y = malloc1d(nSH_order*sizeof(float));
304  grid_dirs_deg = (float*)(&__Tdesign_degree_30_dirs_deg[0][0]);
305  for(ng=0; ng<nGrid_dirs; ng++){
306  azi_incl[0] = grid_dirs_deg[ng*2]*M_PI/180.0f;
307  azi_incl[1] = M_PI/2.0f-grid_dirs_deg[ng*2+1]*M_PI/180.0f;
308  getSHreal(n, (float*)azi_incl, 1, Y);
309  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasTrans, nLoudspeakers, 1, nSH_order, 1.0f,
310  pars->M_dec[d][n-1], nSH_order,
311  Y, nSH_order, 0.0f,
312  g, 1);
313  a[ng] = e[ng] = 0.0f;
314  for(i=0; i<nLoudspeakers; i++){
315  a[ng] += g[i];
316  e[ng] += powf(g[i], 2.0f);
317  }
318  }
319 
320  /* determine the order+decoder dependent normalisation factor for energy+amplitude preserving decoding */
321  a_avg[n-1] = e_avg[n-1] = 0.0f;
322  for(ng=0; ng<nGrid_dirs; ng++){
323  a_avg[n-1] += a[ng];
324  e_avg[n-1] += e[ng];
325  }
326  a_avg[n-1] /= (float)nGrid_dirs;
327  e_avg[n-1] /= (float)nGrid_dirs;
328  pars->M_norm[d][n-1][0] = 1.0f/(a_avg[n-1]+2.23e-6f); /* use this to preserve omni amplitude */
329  pars->M_norm[d][n-1][1] = sqrtf(1.0f/(e_avg[n-1]+2.23e-6f)); /* use this to preserve omni energy */
330  free(a_n);
331  free(Y);
332 
333  /* remove virtual loudspeakers from the decoder */
334  if (pData->loudpkrs_nDims == 2){
335  pars->M_dec[d][n-1] = realloc1d(pars->M_dec[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float));
336  pars->M_dec_cmplx[d][n-1] = realloc1d(pars->M_dec_cmplx[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float_complex));
337  pars->M_dec_maxrE[d][n-1] = realloc1d(pars->M_dec_maxrE[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float));
338  pars->M_dec_cmplx_maxrE[d][n-1] = realloc1d(pars->M_dec_cmplx_maxrE[d][n-1], pData->nLoudpkrs * nSH_order * sizeof(float_complex));
339  }
340  }
341  free(M_dec_tmp);
342  }
343 
344  /* update order */
345  pData->masterOrder = pData->new_masterOrder;
346 
347  /* Binaural-related initialisations */
348  if(pData->reinit_hrtfsFLAG){
349  strcpy(pData->progressBarText,"Computing VBAP gain table");
350  pData->progressBar0_1 = 0.4f;
351 
352  /* load sofa file or load default hrir data */
353 #ifdef SAF_ENABLE_SOFA_READER_MODULE
354  if(!pData->useDefaultHRIRsFLAG && pars->sofa_filepath!=NULL){
355  /* Load SOFA file */
356  error = saf_sofa_open(&sofa, pars->sofa_filepath);
357 
358  /* Load defaults instead */
359  if(error!=SAF_SOFA_OK || sofa.nReceivers!=NUM_EARS){
360  pData->useDefaultHRIRsFLAG = 1;
361  saf_print_warning("Unable to load the specified SOFA file, or it contained something other than 2 channels. Using default HRIR data instead.");
362  }
363  else{
364  /* Copy SOFA data */
365  pars->hrir_fs = (int)sofa.DataSamplingRate;
366  pars->hrir_len = sofa.DataLengthIR;
367  pars->N_hrir_dirs = sofa.nSources;
368  pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
369  memcpy(pars->hrirs, sofa.DataIR, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
370  pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
371  cblas_scopy(pars->N_hrir_dirs, sofa.SourcePosition, 3, pars->hrir_dirs_deg, 2); /* azi */
372  cblas_scopy(pars->N_hrir_dirs, &sofa.SourcePosition[1], 3, &pars->hrir_dirs_deg[1], 2); /* elev */
373  }
374 
375  /* Clean-up */
376  saf_sofa_close(&sofa);
377  }
378 #else
379  pData->useDefaultHRIRsFLAG = 1; /* Can only load the default HRIR data */
380 #endif
381  if(pData->useDefaultHRIRsFLAG){
382  /* Copy default HRIR data */
383  pars->hrir_fs = __default_hrir_fs;
384  pars->hrir_len = __default_hrir_len;
385  pars->N_hrir_dirs = __default_N_hrir_dirs;
386  pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
387  memcpy(pars->hrirs, (float*)__default_hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
388  pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
389  memcpy(pars->hrir_dirs_deg, (float*)__default_hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
390  }
391 
392  /* estimate the ITDs for each HRIR */
393  pars->itds_s = realloc1d(pars->itds_s, pars->N_hrir_dirs*sizeof(float));
394  estimateITDs(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, pars->hrir_fs, pars->itds_s);
395 
396  /* generate VBAP gain table for the hrir_dirs */
397  hrtf_vbap_gtable = NULL;
398  pars->hrtf_vbapTableRes[0] = 2; /* azimuth resolution in degrees */
399  pars->hrtf_vbapTableRes[1] = 5; /* elevation resolution in degrees */
400  generateVBAPgainTable3D(pars->hrir_dirs_deg, pars->N_hrir_dirs, pars->hrtf_vbapTableRes[0], pars->hrtf_vbapTableRes[1], 1, 0, 0.0f,
401  &hrtf_vbap_gtable, &(pars->N_hrtf_vbap_gtable), &(pars->hrtf_nTriangles));
402  if(hrtf_vbap_gtable==NULL){
403  /* if generating vbap gain tabled failed, re-calculate with default HRIR set (which is known to triangulate correctly) */
404  pData->useDefaultHRIRsFLAG = 1;
405  ambi_dec_initCodec(hAmbi);
406  }
407 
408  /* compress VBAP table (i.e. remove the zero elements) */
409  pars->hrtf_vbap_gtableComp = realloc1d(pars->hrtf_vbap_gtableComp, pars->N_hrtf_vbap_gtable * 3 * sizeof(float));
410  pars->hrtf_vbap_gtableIdx = realloc1d(pars->hrtf_vbap_gtableIdx, pars->N_hrtf_vbap_gtable * 3 * sizeof(int));
411  compressVBAPgainTable3D(hrtf_vbap_gtable, pars->N_hrtf_vbap_gtable, pars->N_hrir_dirs, pars->hrtf_vbap_gtableComp, pars->hrtf_vbap_gtableIdx);
412 
413  /* convert hrirs to filterbank coefficients */
414  strcpy(pData->progressBarText,"Preparing HRIRs");
415  pData->progressBar0_1 = 0.85f;
416  pars->hrtf_fb = realloc1d(pars->hrtf_fb, HYBRID_BANDS * NUM_EARS * (pars->N_hrir_dirs)*sizeof(float_complex));
417  HRIRs2HRTFs_afSTFT(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, HOP_SIZE, 0, 1, pars->hrtf_fb);
418  /* HRIR pre-processing */
419  if(pData->enableHRIRsPreProc){
420  /* get integration weights */
421  strcpy(pData->progressBarText,"Applying HRIR Pre-Processing");
422  pData->progressBar0_1 = 0.95f;
423  if(pars->N_hrir_dirs<=3600){
424  pars->weights = realloc1d(pars->weights, pars->N_hrir_dirs*sizeof(float));
425  getVoronoiWeights(pars->hrir_dirs_deg, pars->N_hrir_dirs, 0, pars->weights);
426  }
427  else{
428  pars->weights = realloc1d(pars->weights, pars->N_hrir_dirs*sizeof(float));
429  for(int idx=0; idx < pars->N_hrir_dirs; idx++)
430  pars->weights[idx] = 4.f*SAF_PI / (float)pars->N_hrir_dirs;
431  }
432  diffuseFieldEqualiseHRTFs(pars->N_hrir_dirs, pars->itds_s, pData->freqVector, HYBRID_BANDS, pars->weights, 1, 0, pars->hrtf_fb);
433  }
434 
435  /* calculate magnitude responses */
436  pars->hrtf_fb_mag = realloc1d(pars->hrtf_fb_mag, HYBRID_BANDS*NUM_EARS*(pars->N_hrir_dirs)*sizeof(float));
437  for(i=0; i<HYBRID_BANDS*NUM_EARS* (pars->N_hrir_dirs); i++)
438  pars->hrtf_fb_mag[i] = cabsf(pars->hrtf_fb[i]);
439 
440  /* clean-up */
441  free(hrtf_vbap_gtable);
442  pData->reinit_hrtfsFLAG = 0;
443  }
444 
445  /* done! */
446  strcpy(pData->progressBarText,"Done!");
447  pData->progressBar0_1 = 1.0f;
448  pData->codecStatus = CODEC_STATUS_INITIALISED;
449 
450  free(g);
451  free(a);
452  free(e);
453 }

457  void * const hAmbi,
458  const float *const * inputs,
459  float ** const outputs,
460  int nInputs,
461  int nOutputs,
462  int nSamples
463 )
464 {
465  ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
466  ambi_dec_codecPars* pars = pData->pars;
467  int ch, ear, i, band, orderBand, nSH_band, decIdx, nSH;
468  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
469 
470  /* local copies of user parameters */
471  int nLoudspeakers, binauraliseLS, masterOrder;
472  int orderPerBand[HYBRID_BANDS], rE_WEIGHT[NUM_DECODERS];
473  float transitionFreq;
474  AMBI_DEC_DIFFUSE_FIELD_EQ_APPROACH diffEQmode[NUM_DECODERS];
475  NORM_TYPES norm;
476  CH_ORDER chOrdering;
477  masterOrder = pData->masterOrder;
478  nSH = ORDER2NSH(masterOrder);
479  nLoudspeakers = pData->nLoudpkrs;
480  memcpy(orderPerBand, pData->orderPerBand, HYBRID_BANDS*sizeof(int));
481  transitionFreq = pData->transitionFreq;
482  memcpy(diffEQmode, pData->diffEQmode, NUM_DECODERS*sizeof(int));
483  binauraliseLS = pData->binauraliseLS;
484  norm = pData->norm;
485  chOrdering = pData->chOrdering;
486  memcpy(rE_WEIGHT, pData->rE_WEIGHT, NUM_DECODERS*sizeof(int));
487 
488  /* Process frame */
489  if (nSamples == AMBI_DEC_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
490  pData->procStatus = PROC_STATUS_ONGOING;
491 
492  /* Load time-domain data */
493  for(i=0; i < SAF_MIN(nSH, nInputs); i++)
494  utility_svvcopy(inputs[i], AMBI_DEC_FRAME_SIZE, pData->SHFrameTD[i]);
495  for(; i<nSH; i++)
496  memset(pData->SHFrameTD[i], 0, AMBI_DEC_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
497 
498  /* account for channel order convention */
499  switch(chOrdering){
500  case CH_ACN: /* already ACN, do nothing */ break; /* Otherwise, convert to ACN... */
501  case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHFrameTD), masterOrder, AMBI_DEC_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
502  }
503 
504  /* account for input normalisation scheme */
505  switch(norm){
506  case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
507  case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), masterOrder, AMBI_DEC_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
508  case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), masterOrder, AMBI_DEC_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
509  }
510 
511  /* Apply time-frequency transform (TFT) */
512  afSTFT_forward_knownDimensions(pData->hSTFT, pData->SHFrameTD, AMBI_DEC_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTF);
513 
514  /* Decode to loudspeaker set-up */
515  memset(FLATTEN3D(pData->outputframeTF), 0, HYBRID_BANDS*MAX_NUM_LOUDSPEAKERS*TIME_SLOTS*sizeof(float_complex));
516  for(band=0; band<HYBRID_BANDS; band++){
517  orderBand = SAF_MAX(SAF_MIN(orderPerBand[band], masterOrder),1);
518  nSH_band = (orderBand+1)*(orderBand+1);
519 
520  /* There is a different decoder for low (0) and high (1) frequencies, and for max_rE weights enabled/disabled */
521  decIdx = pData->freqVector[band] < transitionFreq ? 0 : 1;
522  if(rE_WEIGHT[decIdx]){
523  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLoudspeakers, TIME_SLOTS, nSH_band, &calpha,
524  pars->M_dec_cmplx_maxrE[decIdx][orderBand-1], nSH_band,
525  FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
526  FLATTEN2D(pData->outputframeTF[band]), TIME_SLOTS);
527  }
528  else{
529  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nLoudspeakers, TIME_SLOTS, nSH_band, &calpha,
530  pars->M_dec_cmplx[decIdx][orderBand-1], nSH_band,
531  FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
532  FLATTEN2D(pData->outputframeTF[band]), TIME_SLOTS);
533  }
534 
535  /* Apply scaling to preserve either the amplitude or energy when the decododing orders are different over frequency */
536  cblas_sscal(/*re+im*/2*nLoudspeakers*TIME_SLOTS, pars->M_norm[decIdx][orderBand-1][diffEQmode[decIdx]==AMPLITUDE_PRESERVING ? 0 : 1],
537  (float*)FLATTEN2D(pData->outputframeTF[band]), 1);
538  }
539 
540  /* Binauralise the loudspeaker signals */
541  if(binauraliseLS){
542  /* Initialise the binaural buffer with zeros */
543  memset(FLATTEN3D(pData->binframeTF), 0, HYBRID_BANDS*NUM_EARS*TIME_SLOTS * sizeof(float_complex));
544 
545  /* Convolve each loudspeaker signals with the respective HRTFs */
546  for (ch = 0; ch < nLoudspeakers; ch++) {
547  if(pData->recalc_hrtf_interpFLAG[ch]){
548  /* Re-compute the interpolated HRTF (only if loudspeaker direction changed) */
549  ambi_dec_interpHRTFs(hAmbi, pData->loudpkrs_dirs_deg[ch][0], pData->loudpkrs_dirs_deg[ch][1], pars->hrtf_interp[ch]);
550  pData->recalc_hrtf_interpFLAG[ch] = 0;
551  }
552 
553  /* Convolve this loudspeaker channel with the interpolated HRTF, and add it to the binaural buffer */
554  for (band = 0; band < HYBRID_BANDS; band++)
555  for (ear = 0; ear < NUM_EARS; ear++)
556  cblas_caxpy(TIME_SLOTS, &pars->hrtf_interp[ch][band][ear], pData->outputframeTF[band][ch], 1, pData->binframeTF[band][ear], 1);
557  }
558 
559  /* Scale by sqrt(number of loudspeakers) */
560  cblas_sscal(/*re+im*/2*HYBRID_BANDS*NUM_EARS*TIME_SLOTS, 1.0f/sqrtf((float)nLoudspeakers), (float*)FLATTEN3D(pData->binframeTF), 1);
561  }
562 
563  /* inverse-TFT */
564  afSTFT_backward_knownDimensions(pData->hSTFT, binauraliseLS ? pData->binframeTF : pData->outputframeTF,
565  AMBI_DEC_FRAME_SIZE, binauraliseLS ? NUM_EARS : MAX_NUM_LOUDSPEAKERS, TIME_SLOTS, pData->outputFrameTD);
566 
567  /* Copy to output buffer */
568  for(ch = 0; ch < SAF_MIN(binauraliseLS==1 ? NUM_EARS : nLoudspeakers, nOutputs); ch++)
569  utility_svvcopy(pData->outputFrameTD[ch], AMBI_DEC_FRAME_SIZE, outputs[ch]);
570  for (; ch < nOutputs; ch++)
571  memset(outputs[ch], 0, AMBI_DEC_FRAME_SIZE*sizeof(float));
572  }
573  else
574  for (ch=0; ch < nOutputs; ch++)
575  memset(outputs[ch], 0, AMBI_DEC_FRAME_SIZE*sizeof(float));
576 
577  pData->procStatus = PROC_STATUS_NOT_ONGOING;
578 }

709 void ambi_dec_setSourcePreset(void* const hAmbi, int newPresetID)
710 {
711  ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
712  int band, rangeIdx, curOrder, reverse;
713 
714  rangeIdx = 0;
715  curOrder = 1;
716  reverse = 0;
717  switch(newPresetID){
718  /* Ideal spherical harmonics will have SH_ORDER at all frequencies */
719  case MIC_PRESET_IDEAL:
720  for(band=0; band<HYBRID_BANDS; band++)
721  pData->orderPerBand[band] = pData->masterOrder;
722  break;
723 
724  /* For real microphone arrays, the maximum usable spherical harmonic order will depend on frequency */
725  case MIC_PRESET_ZYLIA:
726  for(band=0; band<HYBRID_BANDS; band++){
727  if(rangeIdx<2*(__Zylia_maxOrder-1)){
728  if(pData->freqVector[band]>__Zylia_freqRange[rangeIdx]){
729  if(!reverse)
730  curOrder++;
731  else
732  curOrder--;
733  reverse = (curOrder == __Zylia_maxOrder) || (reverse) ? 1 : 0;
734  rangeIdx++;
735  }
736  }
737  pData->orderPerBand[band] = SAF_MIN(pData->masterOrder,curOrder);
738  }
739  break;
740 
741  case MIC_PRESET_EIGENMIKE32:
742  for(band=0; band<HYBRID_BANDS; band++){
743  if(rangeIdx<2*(__Eigenmike32_maxOrder-1)){
744  if(pData->freqVector[band]>__Eigenmike32_freqRange[rangeIdx]){
745  if(!reverse)
746  curOrder++;
747  else
748  curOrder--;
749  reverse = (curOrder == __Eigenmike32_maxOrder) || (reverse) ? 1 : 0;
750  rangeIdx++;
751  }
752  }
753  pData->orderPerBand[band] = SAF_MIN(pData->masterOrder,curOrder);
754  }
755  break;
756 
757  case MIC_PRESET_DTU_MIC:
758  for(band=0; band<HYBRID_BANDS; band++){
759  if(rangeIdx<2*(__DTU_mic_maxOrder-1)){
760  if(pData->freqVector[band]>__DTU_mic_freqRange[rangeIdx]){
761  if(!reverse)
762  curOrder++;
763  else
764  curOrder--;
765  reverse = (curOrder == __DTU_mic_maxOrder) || (reverse) ? 1 : 0;
766  rangeIdx++;
767  }
768  }
769  pData->orderPerBand[band] = SAF_MIN(pData->masterOrder,curOrder);
770  }
771  break;
772  }
773 }
