58 void ambi_dec_interpHRTFs
59 (
60  void* const hAmbi,
61  float azimuth_deg,
62  float elevation_deg,
63  float_complex h_intrp[HYBRID_BANDS][NUM_EARS]
64 )
65 {
66  ambi_dec_data *pData = (ambi_dec_data*)(hAmbi);
67  ambi_dec_codecPars* pars = pData->pars;
68  int i, band;
69  int aziIndex, elevIndex, N_azi, idx3d;
70  float_complex ipd;
71  float aziRes, elevRes, weights[1][3], itds3[3], itdInterp[1];
72  float magnitudes3[HYBRID_BANDS][3][NUM_EARS], magInterp[HYBRID_BANDS][NUM_EARS];
73 
74  /* find closest pre-computed VBAP direction */
75  aziRes = (float)pars->hrtf_vbapTableRes[0];
76  elevRes = (float)pars->hrtf_vbapTableRes[1];
77  N_azi = (int)(360.0f / aziRes + 0.5f) + 1;
78  aziIndex = (int)(matlab_fmodf(azimuth_deg + 180.0f, 360.0f) / aziRes + 0.5f);
79  elevIndex = (int)((elevation_deg + 90.0f) / elevRes + 0.5f);
80  idx3d = elevIndex * N_azi + aziIndex;
81  for (i = 0; i < 3; i++)
82  weights[0][i] = pars->hrtf_vbap_gtableComp[idx3d*3 + i];
83 
84  /* retrieve the 3 itds and hrtf magnitudes */
85  for (i = 0; i < 3; i++) {
86  itds3[i] = pars->itds_s[pars->hrtf_vbap_gtableIdx[idx3d*3+i]];
87  for (band = 0; band < HYBRID_BANDS; band++) {
88  magnitudes3[band][i][0] = pars->hrtf_fb_mag[band*NUM_EARS*(pars->N_hrir_dirs) + 0*(pars->N_hrir_dirs) + pars->hrtf_vbap_gtableIdx[idx3d*3+i]];
89  magnitudes3[band][i][1] = pars->hrtf_fb_mag[band*NUM_EARS*(pars->N_hrir_dirs) + 1*(pars->N_hrir_dirs) + pars->hrtf_vbap_gtableIdx[idx3d*3+i]];
90  }
91  }
92 
93  /* interpolate hrtf magnitudes and itd seperately */
94  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 1, 1, 3, 1,
95  (float*)weights, 3,
96  (float*)itds3, 1, 0,
97  (float*)itdInterp, 1);
98  for (band = 0; band < HYBRID_BANDS; band++) {
99  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, 1, 2, 3, 1,
100  (float*)weights, 3,
101  (float*)magnitudes3[band], 2, 0,
102  (float*)magInterp[band], 2);
103  }
104 
105  /* reintroduce the interaural phase difference per band */
106  for (band = 0; band < HYBRID_BANDS; band++) {
107  if(pData->freqVector[band]<1.5e3f)
108  ipd = cmplxf(0.0f, (matlab_fmodf(2.0f*SAF_PI*(pData->freqVector[band]) * itdInterp[0] + SAF_PI, 2.0f*SAF_PI) - SAF_PI)/2.0f);
109  else
110  ipd = cmplxf(0.0f, 0.0f);
111  h_intrp[band][0] = ccmulf(cmplxf(magInterp[band][0], 0.0f), cexpf(ipd));
112  h_intrp[band][1] = ccmulf(cmplxf(magInterp[band][1], 0.0f), conjf(cexpf(ipd)));
113  }
114 }
