35 void powermap_create
36 (
37  void ** const phPm
38 )
39 {
40  powermap_data* pData = (powermap_data*)malloc1d(sizeof(powermap_data));
41  *phPm = (void*)pData;
42  int n, i, band;
43 
44  /* Default user parameters */
45  pData->masterOrder = pData->new_masterOrder = SH_ORDER_FIRST;
46  for(band=0; band<HYBRID_BANDS; band++){
47  pData->analysisOrderPerBand[band] = pData->masterOrder;
48  pData->pmapEQ[band] = 1.0f;
49  }
50  pData->covAvgCoeff = 0.0f;
51  pData->pmapAvgCoeff = 0.666f;
52  pData->nSources = 1;
53  pData->pmap_mode = PM_MODE_MUSIC;
54  pData->HFOVoption = HFOV_360;
55  pData->aspectRatioOption = ASPECT_RATIO_2_1;
56  pData->chOrdering = CH_ACN;
57  pData->norm = NORM_SN3D;
58 
59  afSTFT_create(&(pData->hSTFT), MAX_NUM_SH_SIGNALS, 0, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
60  pData->SHframeTD = (float**)malloc2d(MAX_NUM_SH_SIGNALS, POWERMAP_FRAME_SIZE, sizeof(float));
61  pData->SHframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
62 
63  /* codec data */
64  pData->pars = (powermap_codecPars*)malloc1d(sizeof(powermap_codecPars));
65  powermap_codecPars* pars = pData->pars;
66  pars->interp_dirs_deg = NULL;
67  for(n=0; n<MAX_SH_ORDER; n++){
68  pars->Y_grid[n] = NULL;
69  pars->Y_grid_cmplx[n] = NULL;
70  }
71  pars->interp_table = NULL;
72 
73  /* internal */
74  pData->progressBar0_1 = 0.0f;
75  pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
76  strcpy(pData->progressBarText,"");
77  pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
78  pData->procStatus = PROC_STATUS_NOT_ONGOING;
79  pData->dispWidth = 140;
80 
81  /* display */
82  pData->pmap = NULL;
83  pData->prev_pmap = NULL;
84  for(i=0; i<NUM_DISP_SLOTS; i++)
85  pData->pmap_grid[i] = NULL;
86  pData->pmapReady = 0;
87  pData->recalcPmap = 1;
88 
89  /* set FIFO buffer */
90  pData->FIFO_idx = 0;
91  memset(pData->inFIFO, 0, MAX_NUM_SH_SIGNALS*POWERMAP_FRAME_SIZE*sizeof(float));
92 }

94 void powermap_destroy
95 (
96  void ** const phPm
97 )
98 {
99  powermap_data *pData = (powermap_data*)(*phPm);
100  powermap_codecPars* pars;
101  int i;
102 
103  if (pData != NULL) {
104  /* not safe to free memory during intialisation/processing loop */
105  while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
106  pData->procStatus == PROC_STATUS_ONGOING){
107  SAF_SLEEP(10);
108  }
109 
110  pars = pData->pars;
111 
112  /* free afSTFT and buffers */
113  afSTFT_destroy(&(pData->hSTFT));
114  free(pData->SHframeTD);
115  free(pData->SHframeTF);
116 
117  free(pData->pmap);
118  free(pData->prev_pmap);
119  for(i=0; i<NUM_DISP_SLOTS; i++)
120  free(pData->pmap_grid[i]);
121  free(pars->interp_dirs_deg);
122  for(i=0; i<MAX_SH_ORDER; i++){
123  free(pars->Y_grid[i]);
124  free(pars->Y_grid_cmplx[i]);
125  }
126  free(pars->interp_table);
127  free(pData->pars);
128  free(pData->progressBarText);
129  free(pData);
130  pData = NULL;
131  }

156 void powermap_initCodec
157 (
158  void* const hPm
159 )
160 {
161  powermap_data *pData = (powermap_data*)(hPm);
162 
163  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
164  return; /* re-init not required, or already happening */
165  while (pData->procStatus == PROC_STATUS_ONGOING){
166  /* re-init required, but we need to wait for the current processing loop to end */
167  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
168  SAF_SLEEP(10);
169  }
170 
171  /* for progress bar */
172  pData->codecStatus = CODEC_STATUS_INITIALISING;
173  strcpy(pData->progressBarText,"Initialising");
174  pData->progressBar0_1 = 0.0f;
175 
176  powermap_initTFT(hPm);
177  powermap_initAna(hPm);
178 
179  /* done! */
180  strcpy(pData->progressBarText,"Done!");
181  pData->progressBar0_1 = 1.0f;
182  pData->codecStatus = CODEC_STATUS_INITIALISED;
183 }

185 void powermap_analysis
186 (
187  void * const hPm,
188  const float *const * inputs,
189  int nInputs,
190  int nSamples,
191  int isPlaying
192 )
193 {
194  powermap_data *pData = (powermap_data*)(hPm);
195  powermap_codecPars* pars = pData->pars;
196  int s, i, j, ch, band, nSH_order, order_band, nSH_maxOrder, maxOrder;
197  float C_grp_trace, pmapEQ_band;
198  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
199  float_complex new_Cx[MAX_NUM_SH_SIGNALS*MAX_NUM_SH_SIGNALS];
200  float_complex C_grp[MAX_NUM_SH_SIGNALS*MAX_NUM_SH_SIGNALS];
201 
202  /* local parameters */
203  int analysisOrderPerBand[HYBRID_BANDS];
204  int nSources, masterOrder, nSH;
205  float covAvgCoeff, pmapAvgCoeff;
206  float pmapEQ[HYBRID_BANDS];
207  NORM_TYPES norm;
208  CH_ORDER chOrdering;
209  POWERMAP_MODES pmap_mode;
210  memcpy(analysisOrderPerBand, pData->analysisOrderPerBand, HYBRID_BANDS*sizeof(int));
211  memcpy(pmapEQ, pData->pmapEQ, HYBRID_BANDS*sizeof(float));
212  norm = pData->norm;
213  chOrdering = pData->chOrdering;
214  nSources = pData->nSources;
215  covAvgCoeff = SAF_MIN(pData->covAvgCoeff, MAX_COV_AVG_COEFF);
216  pmapAvgCoeff = pData->pmapAvgCoeff;
217  pmap_mode = pData->pmap_mode;
218  masterOrder = pData->masterOrder;
219  nSH = (masterOrder+1)*(masterOrder+1);
220 
221  /* Loop over all samples */
222  for(s=0; s<nSamples; s++){
223  /* Load input signals into inFIFO buffer */
224  for(ch=0; ch<SAF_MIN(nInputs,nSH); ch++)
225  pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
226  for(; ch<nSH; ch++) /* Zero any channels that were not given */
227  pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
228 
229  /* Increment buffer index */
230  pData->FIFO_idx++;
231 
232  /* Process frame if inFIFO is full and codec is ready for it */
233  if (pData->FIFO_idx >= POWERMAP_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) && isPlaying ) {
234  pData->FIFO_idx = 0;
235  pData->procStatus = PROC_STATUS_ONGOING;
236 
237  /* Load time-domain data */
238  for(ch=0; ch<nSH; ch++)
239  memcpy(pData->SHframeTD[ch], pData->inFIFO[ch], POWERMAP_FRAME_SIZE*sizeof(float));
240 
241  /* account for input channel order */
242  switch(chOrdering){
243  case CH_ACN: /* already ACN */ break; /* Otherwise, convert to ACN... */
244  case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHframeTD), masterOrder, POWERMAP_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
245  }
246 
247  /* account for input normalisation scheme */
248  switch(norm){
249  case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
250  case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHframeTD), masterOrder, POWERMAP_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
251  case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHframeTD), masterOrder, POWERMAP_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
252  }
253 
254  /* apply the time-frequency transform */
255  afSTFT_forward_knownDimensions(pData->hSTFT, pData->SHframeTD, POWERMAP_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTF);
256 
257  /* Update covarience matrix per band */
258  for(band=0; band<HYBRID_BANDS; band++){
259  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, nSH, nSH, TIME_SLOTS, &calpha,
260  FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS,
261  FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
262  new_Cx, nSH);
263 
264  /* average over time */
265  cblas_sscal(nSH*nSH*2, covAvgCoeff, (float*)pData->Cx[band], 1);
266  cblas_saxpy(nSH*nSH*2, 1.0f-covAvgCoeff, (float*)new_Cx, 1, (float*)pData->Cx[band], 1);
267  }
268 
269  /* update the powermap */
270  if(pData->recalcPmap==1){
271  pData->recalcPmap = 0;
272  pData->pmapReady = 0;
273 
274  /* determine maximum analysis order */
275  maxOrder = 1;
276  for(i=0; i<HYBRID_BANDS; i++)
277  maxOrder = SAF_MAX(maxOrder, SAF_MIN(analysisOrderPerBand[i], masterOrder));
278  nSH_maxOrder = (maxOrder+1)*(maxOrder+1);
279 
280  /* group covarience matrices */
281  memset(C_grp, 0, nSH_maxOrder*nSH_maxOrder*sizeof(float_complex));
282  for (band=0; band<HYBRID_BANDS; band++){
283  order_band = SAF_MAX(SAF_MIN(pData->analysisOrderPerBand[band], masterOrder),1);
284  nSH_order = (order_band+1)*(order_band+1);
285  pmapEQ_band = SAF_MIN(SAF_MAX(pmapEQ[band], 0.0f), 2.0f);
286  for(i=0; i<nSH_order; i++)
287  for(j=0; j<nSH_order; j++)
288  C_grp[i*nSH_maxOrder+j] = ccaddf(C_grp[i*nSH_maxOrder+j], crmulf(pData->Cx[band][i*nSH+j], 1e3f*pmapEQ_band));
289  }
290 
291  /* generate powermap */
292  C_grp_trace = 0.0f;
293  for(i=0; i<nSH_maxOrder; i++)
294  C_grp_trace+=crealf(C_grp[i*nSH_maxOrder+ i]);
295  switch(pmap_mode){
296  default:
297  case PM_MODE_PWD:
298  generatePWDmap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], pars->grid_nDirs, pData->pmap);
299  break;
300 
301  case PM_MODE_MVDR:
302  if(C_grp_trace>1e-8f)
303  generateMVDRmap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], pars->grid_nDirs, 8.0f, pData->pmap, NULL);
304  else
305  memset(pData->pmap, 0, pars->grid_nDirs*sizeof(float));
306  break;
307 
308  case PM_MODE_CROPAC_LCMV:
309  if(C_grp_trace>1e-8f)
310  generateCroPaCLCMVmap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], pars->grid_nDirs, 8.0f, 0.0f, pData->pmap);
311  else
312  memset(pData->pmap, 0, pars->grid_nDirs*sizeof(float));
313  break;
314 
315  case PM_MODE_MUSIC:
316  if(C_grp_trace>1e-8f)
317  generateMUSICmap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], nSources, pars->grid_nDirs, 0, pData->pmap);
318  else
319  memset(pData->pmap, 0, pars->grid_nDirs*sizeof(float));
320  break;
321 
322  case PM_MODE_MUSIC_LOG:
323  if(C_grp_trace>1e-8f)
324  generateMUSICmap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], nSources, pars->grid_nDirs, 1, pData->pmap);
325  else
326  memset(pData->pmap, 0, pars->grid_nDirs*sizeof(float));
327  break;
328 
329  case PM_MODE_MINNORM:
330  if(C_grp_trace>1e-8f)
331  generateMinNormMap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], nSources, pars->grid_nDirs, 0, pData->pmap);
332  else
333  memset(pData->pmap, 0, pars->grid_nDirs*sizeof(float));
334  break;
335 
336  case PM_MODE_MINNORM_LOG:
337  if(C_grp_trace>1e-8f)
338  generateMinNormMap(maxOrder, (float_complex*)C_grp, pars->Y_grid_cmplx[maxOrder-1], nSources, pars->grid_nDirs, 1, pData->pmap);
339  else
340  memset(pData->pmap, 0, pars->grid_nDirs*sizeof(float));
341  break;
342  }
343 
344  /* average powermap over time */
345  for(i=0; i<pars->grid_nDirs; i++)
346  pData->pmap[i] = (1.0f-pmapAvgCoeff) * (pData->pmap[i] )+ pmapAvgCoeff * (pData->prev_pmap[i]);
347  utility_svvcopy(pData->pmap, pars->grid_nDirs, pData->prev_pmap);
348 
349  /* interpolate powermap */
350  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->interp_nDirs, 1, pars->grid_nDirs, 1.0f,
351  pars->interp_table, pars->grid_nDirs,
352  pData->pmap, 1, 0.0f,
353  pData->pmap_grid[pData->dispSlotIdx], 1);
354 
355  /* ascertain minimum and maximum values for powermap colour scaling */
356  int ind;
357  utility_siminv(pData->pmap_grid[pData->dispSlotIdx], pars->interp_nDirs, &ind);
358  pData->pmap_grid_minVal = pData->pmap_grid[pData->dispSlotIdx][ind];
359  utility_simaxv(pData->pmap_grid[pData->dispSlotIdx], pars->interp_nDirs, &ind);
360  pData->pmap_grid_maxVal = pData->pmap_grid[pData->dispSlotIdx][ind];
361 
362  /* normalise the powermap to 0..1 */
363  for(i=0; i<pars->interp_nDirs; i++)
364  pData->pmap_grid[pData->dispSlotIdx][i] = (pData->pmap_grid[pData->dispSlotIdx][i]-pData->pmap_grid_minVal)/(pData->pmap_grid_maxVal-pData->pmap_grid_minVal+1e-11f);
365 
366  /* signify that the powermap in current slot is ready for plotting */
367  pData->dispSlotIdx++;
368  if(pData->dispSlotIdx>=NUM_DISP_SLOTS)
369  pData->dispSlotIdx = 0;
370  pData->pmapReady = 1;
371  }
372  }
373  else if(pData->FIFO_idx >= POWERMAP_FRAME_SIZE){
374  /* reset FIFO_idx if codec was not ready */
375  pData->FIFO_idx = 0;
376  }
377  }
378 
379  pData->procStatus = PROC_STATUS_NOT_ONGOING;
380 }

424 void powermap_setSourcePreset(void* const hPm, int newPresetID)
425 {
426  powermap_data *pData = (powermap_data*)(hPm);
427  int band, rangeIdx, curOrder, reverse;
428 
429  rangeIdx = 0;
430  curOrder = 1;
431  reverse = 0;
432  switch(newPresetID){
433  case MIC_PRESET_IDEAL:
434  /* Ideal SH should have maximum order per frequency */
435  for(band=0; band<HYBRID_BANDS; band++)
436  pData->analysisOrderPerBand[band] = pData->new_masterOrder;
437  break;
438 
439  /* In the case of real microphone arrays, the analysis order should be frequency dependent
440  * and the frequencies above the spatial-aliasing limit should be EQ's out. */
441  case MIC_PRESET_ZYLIA:
442  for(band=0; band<HYBRID_BANDS; band++){
443  if(rangeIdx<2*(__Zylia_maxOrder-1)){
444  if(pData->freqVector[band]>__Zylia_freqRange[rangeIdx]){
445  if(!reverse)
446  curOrder++;
447  else
448  curOrder--;
449  reverse = (curOrder == __Zylia_maxOrder) || (reverse) ? 1 : 0;
450  rangeIdx++;
451  }
452  }
453  pData->analysisOrderPerBand[band] = SAF_MIN(pData->new_masterOrder,curOrder);
454  if(pData->freqVector[band] > __Zylia_freqRange[(__Zylia_maxOrder-1)*2 - 1])
455  pData->pmapEQ[band] = 0.0f;
456  }
457  break;
458 
459  case MIC_PRESET_EIGENMIKE32:
460  for(band=0; band<HYBRID_BANDS; band++){
461  if(rangeIdx<2*(__Eigenmike32_maxOrder-1)){
462  if(pData->freqVector[band]>__Eigenmike32_freqRange[rangeIdx]){
463  if(!reverse)
464  curOrder++;
465  else
466  curOrder--;
467  reverse = (curOrder == __Eigenmike32_maxOrder) || (reverse) ? 1 : 0;
468  rangeIdx++;
469  }
470  }
471  pData->analysisOrderPerBand[band] = SAF_MIN(pData->new_masterOrder,curOrder);
472  if(pData->freqVector[band] > __Eigenmike32_freqRange[(__Eigenmike32_maxOrder-1)*2 - 1])
473  pData->pmapEQ[band] = 0.0f;
474  }
475  break;
476 
477  case MIC_PRESET_DTU_MIC:
478  for(band=0; band<HYBRID_BANDS; band++){
479  if(rangeIdx<2*(__DTU_mic_maxOrder-1)){
480  if(pData->freqVector[band]>__DTU_mic_freqRange[rangeIdx]){
481  if(!reverse)
482  curOrder++;
483  else
484  curOrder--;
485  reverse = (curOrder == __DTU_mic_maxOrder) || (reverse) ? 1 : 0;
486  rangeIdx++;
487  }
488  }
489  pData->analysisOrderPerBand[band] = SAF_MIN(pData->new_masterOrder,curOrder);
490  if(pData->freqVector[band] > __DTU_mic_freqRange[(__DTU_mic_maxOrder-1)*2 - 1])
491  pData->pmapEQ[band] = 0.0f;
492  }
493  break;
494  }
495 }

527 void powermap_setChOrder(void* const hPm, int newOrder)
