164 void ambi_bin_initCodec
165 (
166  void* const hAmbi
167 )
168 {
169  ambi_bin_data *pData = (ambi_bin_data*)(hAmbi);
170  ambi_bin_codecPars* pars = pData->pars;
171  int i, j, nSH, order, band;
172 #ifdef SAF_ENABLE_SOFA_READER_MODULE
173  SAF_SOFA_ERROR_CODES error;
174  saf_sofa_container sofa;
175 #endif
176 
177  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
178  return; /* re-init not required, or already happening */
179  while (pData->procStatus == PROC_STATUS_ONGOING){
180  /* re-init required, but we need to wait for the current processing loop to end */
181  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
182  SAF_SLEEP(10);
183  }
184 
185  /* for progress bar */
186  pData->codecStatus = CODEC_STATUS_INITIALISING;
187  strcpy(pData->progressBarText,"Preparing HRIRs");
188  pData->progressBar0_1 = 0.0f;
189 
190  /* (Re)Initialise afSTFT */
191  order = pData->new_order;
192  nSH = (order+1)*(order+1);
193  if(pData->hSTFT==NULL)
194  afSTFT_create(&(pData->hSTFT), nSH, NUM_EARS, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
195  else if(pData->nSH != nSH) {/* Or change the number of channels */
196  afSTFT_channelChange(pData->hSTFT, nSH, NUM_EARS);
197  afSTFT_clearBuffers(pData->hSTFT);
198  }
199  pData->nSH = nSH;
200 
201  if(pData->reinit_hrtfsFLAG){
202  /* load sofa file or default hrir data */
203  strcpy(pData->progressBarText,"Preparing HRIRs");
204  pData->progressBar0_1 = 0.15f;
205  /* load sofa file or load default hrir data */
206 #ifdef SAF_ENABLE_SOFA_READER_MODULE
207  if(!pData->useDefaultHRIRsFLAG && pars->sofa_filepath!=NULL){
208  /* Load SOFA file */
209  error = saf_sofa_open(&sofa, pars->sofa_filepath);
210 
211  /* Load defaults instead */
212  if(error!=SAF_SOFA_OK || sofa.nReceivers!=NUM_EARS){
213  pData->useDefaultHRIRsFLAG = 1;
214  saf_print_warning("Unable to load the specified SOFA file, or it contained something other than 2 channels. Using default HRIR data instead.");
215  }
216  else{
217  /* Copy SOFA data */
218  pars->hrir_fs = (int)sofa.DataSamplingRate;
219  pars->hrir_len = sofa.DataLengthIR;
220  pars->N_hrir_dirs = sofa.nSources;
221  pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
222  memcpy(pars->hrirs, sofa.DataIR, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
223  pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
224  cblas_scopy(pars->N_hrir_dirs, sofa.SourcePosition, 3, pars->hrir_dirs_deg, 2); /* azi */
225  cblas_scopy(pars->N_hrir_dirs, &sofa.SourcePosition[1], 3, &pars->hrir_dirs_deg[1], 2); /* elev */
226  }
227 
228  /* Clean-up */
229  saf_sofa_close(&sofa);
230  }
231 #else
232  pData->useDefaultHRIRsFLAG = 1; /* Can only load the default HRIR data */
233 #endif
234  if(pData->useDefaultHRIRsFLAG){
235  /* Copy default HRIR data */
236  pars->hrir_fs = __default_hrir_fs;
237  pars->hrir_len = __default_hrir_len;
238  pars->N_hrir_dirs = __default_N_hrir_dirs;
239  pars->hrirs = realloc1d(pars->hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
240  memcpy(pars->hrirs, (float*)__default_hrirs, pars->N_hrir_dirs*NUM_EARS*(pars->hrir_len)*sizeof(float));
241  pars->hrir_dirs_deg = realloc1d(pars->hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
242  memcpy(pars->hrir_dirs_deg, (float*)__default_hrir_dirs_deg, pars->N_hrir_dirs*2*sizeof(float));
243  }
244 
245  /* estimate the ITDs for each HRIR */
246  pData->progressBar0_1 = 0.3f;
247  pars->itds_s = realloc1d(pars->itds_s, pars->N_hrir_dirs*sizeof(float));
248  estimateITDs(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, pars->hrir_fs, pars->itds_s);
249 
250  /* convert hrirs to filterbank coefficients */
251  pData->progressBar0_1 = 0.4f;
252  pars->hrtf_fb = realloc1d(pars->hrtf_fb, HYBRID_BANDS * NUM_EARS * (pars->N_hrir_dirs)*sizeof(float_complex));
253  HRIRs2HRTFs_afSTFT(pars->hrirs, pars->N_hrir_dirs, pars->hrir_len, HOP_SIZE, 0, 1, pars->hrtf_fb);
254  /* get integration weights */
255  pData->progressBar0_1 = 0.6f;
256  if(pars->N_hrir_dirs<=1000){
257  pars->weights = realloc1d(pars->weights, pars->N_hrir_dirs*sizeof(float));
258  getVoronoiWeights(pars->hrir_dirs_deg, pars->N_hrir_dirs, 0, pars->weights);
259  }
260  else{
261  free(pars->weights);
262  pars->weights = NULL;
263  }
264  /* HRIR pre-processing */
265  pData->progressBar0_1 = 0.75f;
266  diffuseFieldEqualiseHRTFs(pars->N_hrir_dirs, pars->itds_s, pData->freqVector, HYBRID_BANDS, pars->weights,
267  pData->preProc == HRIR_PREPROC_EQ || pData->preProc == HRIR_PREPROC_ALL ? 1 : 0, /* Apply Diffuse-field EQ? */
268  pData->preProc == HRIR_PREPROC_PHASE || pData->preProc == HRIR_PREPROC_ALL ? 1 : 0, /* Apply phase simplification EQ? */
269  pars->hrtf_fb);
270  pData->reinit_hrtfsFLAG = 0;
271  }
272 
273  /* get new decoder */
274  strcpy(pData->progressBarText,"Computing Decoder");
275  pData->progressBar0_1 = 0.95f;
276  float_complex* decMtx;
277  decMtx = calloc1d(HYBRID_BANDS*NUM_EARS*nSH, sizeof(float_complex));
278  switch(pData->method){
279  default:
280  case DECODING_METHOD_LS:
281  getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
282  BINAURAL_DECODER_LS, order, pData->freqVector, pars->itds_s, pars->weights,
283  pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
284  break;
285  case DECODING_METHOD_LSDIFFEQ:
286  getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
287  BINAURAL_DECODER_LSDIFFEQ, order, pData->freqVector, pars->itds_s, pars->weights,
288  pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
289  break;
290  case DECODING_METHOD_SPR:
291  getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
292  BINAURAL_DECODER_SPR, order, pData->freqVector, pars->itds_s, pars->weights,
293  pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
294  break;
295  case DECODING_METHOD_TA:
296  getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
297  BINAURAL_DECODER_TA, order, pData->freqVector, pars->itds_s, pars->weights,
298  pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
299  break;
300  case DECODING_METHOD_MAGLS:
301  getBinauralAmbiDecoderMtx(pars->hrtf_fb, pars->hrir_dirs_deg, pars->N_hrir_dirs, HYBRID_BANDS,
302  BINAURAL_DECODER_MAGLS, order, pData->freqVector, pars->itds_s, pars->weights,
303  pData->enableDiffuseMatching, pData->enableMaxRE, decMtx);
304  break;
305  }
306 
307  /* Apply Truncation EQ */
308  if(pData->enableTruncationEQ &&
309  pData->method==DECODING_METHOD_LS &&
310  pData->preProc!=HRIR_PREPROC_PHASE &&
311  pData->preProc!=HRIR_PREPROC_ALL)
312  {
313  double *kr;
314  float *w_n, *eqGain;
315  const int order_truncated = order;
316  const int order_target = 42; /* Equalizing diffuse field to 42nd order equivalent. */
317  const float softThreshold = 9.0; /* results in +9 dB max */
318  const double r = 0.085; /* spherical scatterer radius (approx. size of human head) */
319  const int numBands = HYBRID_BANDS;
320  const double c = 343.;
321 
322  /* Prep */
323  kr = malloc1d(numBands * sizeof(double));
324  w_n = calloc1d((order_truncated+1), sizeof(float));
325  eqGain = calloc1d(numBands, sizeof(float));
326  for (int k=0; k<numBands; k++)
327  kr[k] = 2.0*SAF_PId / c * (double)pData->freqVector[k] * r;
328 
329  if (pData->enableMaxRE) {
330  /* maxRE as order weighting */
331  float *maxRECoeffs = malloc1d((order_truncated+1) * sizeof(float));
332  beamWeightsMaxEV(order_truncated, maxRECoeffs);
333  for (int idx_n=0; idx_n<order_truncated+1; idx_n++) {
334  w_n[idx_n] = maxRECoeffs[idx_n];
335  w_n[idx_n] /= sqrtf((float)(2*idx_n+1) / (4.0f*SAF_PI));
336  }
337  float w_0 = w_n[0];
338  for (int idx_n=0; idx_n<order_truncated+1; idx_n++)
339  w_n[idx_n] /= w_0;
340  free(maxRECoeffs);
341  }
342  else {
343  /* just truncation, no tapering */
344  for (int idx_n=0; idx_n<order_truncated+1; idx_n++)
345  w_n[idx_n] = 1.0f;
346  }
347  truncationEQ(w_n, order_truncated, order_target, kr, numBands, softThreshold, eqGain);
348 
349  /* apply to decoding matrix */
350  for (int idxBand=0; idxBand<numBands; idxBand++){
351  for (int idxSH=0; idxSH<pData->nSH; idxSH++){
352  decMtx[idxBand*NUM_EARS*nSH+0*nSH+idxSH] = crmulf(decMtx[idxBand*NUM_EARS*nSH+0*nSH+idxSH], eqGain[idxBand]); /* left ear */
353  decMtx[idxBand*NUM_EARS*nSH+1*nSH+idxSH] = crmulf(decMtx[idxBand*NUM_EARS*nSH+1*nSH+idxSH], eqGain[idxBand]); /* right ear */
354  }
355  }
356 
357  /* clean-up */
358  free(kr);
359  free(w_n);
360  free(eqGain);
361  }
362 
363  /* replace current decoder */
364  memset(pars->M_dec, 0, HYBRID_BANDS*NUM_EARS*MAX_NUM_SH_SIGNALS*sizeof(float_complex));
365  for(band=0; band<HYBRID_BANDS; band++)
366  for(i=0; i<NUM_EARS; i++)
367  for(j=0; j<nSH; j++)
368  pars->M_dec[band][i][j] = decMtx[band*NUM_EARS*nSH + i*nSH + j];
369  free(decMtx);
370 
371  pData->order = order;
372 
373  /* done! */
374  strcpy(pData->progressBarText,"Done!");
375  pData->progressBar0_1 = 1.0f;
376  pData->codecStatus = CODEC_STATUS_INITIALISED;
377 }

379 void ambi_bin_process
380 (
381  void * const hAmbi,
382  const float *const * inputs,
383  float ** const outputs,
384  int nInputs,
385  int nOutputs,
386  int nSamples
387 )
388 {
389  ambi_bin_data *pData = (ambi_bin_data*)(hAmbi);
390  ambi_bin_codecPars* pars = pData->pars;
391  int ch, i, j, band;
392  const float_complex calpha = cmplxf(1.0f,0.0f), cbeta = cmplxf(0.0f, 0.0f);
393  float Rxyz[3][3];
394  float M_rot_tmp[MAX_NUM_SH_SIGNALS*MAX_NUM_SH_SIGNALS];
395 
396  /* local copies of user parameters */
397  int order, nSH, enableRot;
398  NORM_TYPES norm;
399  CH_ORDER chOrdering;
400  norm = pData->norm;
401  chOrdering = pData->chOrdering;
402  order = pData->order;
403  nSH = (order+1)*(order+1);
404  enableRot = pData->enableRotation;
405 
406  /* Process frame */
407  if (nSamples == AMBI_BIN_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
408  pData->procStatus = PROC_STATUS_ONGOING;
409 
410  /* Load time-domain data */
411  for(i=0; i < SAF_MIN(nSH, nInputs); i++)
412  utility_svvcopy(inputs[i], AMBI_BIN_FRAME_SIZE, pData->SHFrameTD[i]);
413  for(; i<nSH; i++)
414  memset(pData->SHFrameTD[i], 0, AMBI_BIN_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
415 
416  /* account for channel order convention */
417  switch(chOrdering){
418  case CH_ACN: /* already in ACN, do nothing */ break; /* Otherwise, convert to ACN... */
419  case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHFrameTD), order, AMBI_BIN_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
420  }
421 
422  /* account for input normalisation scheme */
423  switch(norm){
424  case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
425  case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), order, AMBI_BIN_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
426  case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHFrameTD), order, AMBI_BIN_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
427  }
428 
429  /* Apply time-frequency transform (TFT) */
430  afSTFT_forward_knownDimensions(pData->hSTFT, pData->SHFrameTD, AMBI_BIN_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTF);
431 
432  /* Main processing: */
433  if(order > 0 && enableRot) {
434  /* Apply rotation */
435  if(pData->recalc_M_rotFLAG){
436  /* Compute the new SH rotation matrix */
437  memset(pData->M_rot, 0, MAX_NUM_SH_SIGNALS*MAX_NUM_SH_SIGNALS*sizeof(float_complex));
438  yawPitchRoll2Rzyx(pData->yaw, pData->pitch, pData->roll, pData->useRollPitchYawFlag, Rxyz);
439  getSHrotMtxReal(Rxyz, (float*)M_rot_tmp, order);
440  for (i = 0; i < nSH; i++)
441  for (j = 0; j < nSH; j++)
442  pData->M_rot[i][j] = cmplxf(M_rot_tmp[i*nSH + j], 0.0f);
443 
444  /* Bake the rotation into the decoding matrix */
445  for(band = 0; band < HYBRID_BANDS; band++) {
446  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, NUM_EARS, nSH, nSH, &calpha,
447  pars->M_dec[band], MAX_NUM_SH_SIGNALS,
448  pData->M_rot, MAX_NUM_SH_SIGNALS, &cbeta,
449  pars->M_dec_rot[band], MAX_NUM_SH_SIGNALS);
450  }
451  pData->recalc_M_rotFLAG = 0;
452  }
453  }
454 
455  /* Apply the decoder to go from SH input to binaural output */
456  for(band = 0; band < HYBRID_BANDS; band++) {
457  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, NUM_EARS, TIME_SLOTS, nSH, &calpha,
458  enableRot ? pars->M_dec_rot[band] : pars->M_dec[band], MAX_NUM_SH_SIGNALS,
459  FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS, &cbeta,
460  FLATTEN2D(pData->binframeTF[band]), TIME_SLOTS);
461  }
462 
463  /* inverse-TFT */
464  afSTFT_backward_knownDimensions(pData->hSTFT, pData->binframeTF, AMBI_BIN_FRAME_SIZE, NUM_EARS, TIME_SLOTS, pData->binFrameTD);
465 
466  /* Copy to output */
467  for (ch = 0; ch < SAF_MIN(NUM_EARS, nOutputs); ch++)
468  utility_svvcopy(pData->binFrameTD[ch], AMBI_BIN_FRAME_SIZE, outputs[ch]);
469  for (; ch < nOutputs; ch++)
470  memset(outputs[ch], 0, AMBI_BIN_FRAME_SIZE*sizeof(float));
471  }
472  else
473  for (ch=0; ch < nOutputs; ch++)
474  memset(outputs[ch],0, AMBI_BIN_FRAME_SIZE*sizeof(float));
475 
476  pData->procStatus = PROC_STATUS_NOT_ONGOING;
477 }
