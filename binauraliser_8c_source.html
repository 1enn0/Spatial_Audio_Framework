35 void binauraliser_create
36 (
37  void ** const phBin
38 )
39 {
40  binauraliser_data* pData = (binauraliser_data*)malloc1d(sizeof(binauraliser_data));
41  *phBin = (void*)pData;
42  int ch, dummy;
43 
44  /* user parameters */
45  binauraliser_loadPreset(SOURCE_CONFIG_PRESET_DEFAULT, pData->src_dirs_deg, &(pData->new_nSources), &(dummy)); /*check setStateInformation if you change default preset*/
46  pData->useDefaultHRIRsFLAG = 1; /* pars->sofa_filepath must be valid to set this to 0 */
47  pData->enableHRIRsDiffuseEQ = 1;
48  pData->nSources = pData->new_nSources;
49  pData->interpMode = INTERP_TRI_PS;
50  pData->yaw = 0.0f;
51  pData->pitch = 0.0f;
52  pData->roll = 0.0f;
53  pData->bFlipYaw = 0;
54  pData->bFlipPitch = 0;
55  pData->bFlipRoll = 0;
56  pData->useRollPitchYawFlag = 0;
57  pData->enableRotation = 0;
58 
59  /* time-frequency transform + buffers */
60  pData->hSTFT = NULL;
61  pData->inputFrameTD = (float**)malloc2d(MAX_NUM_INPUTS, BINAURALISER_FRAME_SIZE, sizeof(float));
62  pData->outframeTD = (float**)malloc2d(NUM_EARS, BINAURALISER_FRAME_SIZE, sizeof(float));
63  pData->inputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_INPUTS, TIME_SLOTS, sizeof(float_complex));
64  pData->outputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, NUM_EARS, TIME_SLOTS, sizeof(float_complex));
65 
66  /* hrir data */
67  pData->hrirs = NULL;
68  pData->hrir_dirs_deg = NULL;
69  pData->sofa_filepath = NULL;
70  pData->weights = NULL;
71  pData->N_hrir_dirs = pData->hrir_loaded_len = pData->hrir_runtime_len = 0;
72  pData->hrir_loaded_fs = pData->hrir_runtime_fs = -1; /* unknown */
73 
74  /* vbap (amplitude normalised) */
75  pData->hrtf_vbap_gtableIdx = NULL;
76  pData->hrtf_vbap_gtableComp = NULL;
77  pData->nTriangles = pData->N_hrtf_vbap_gtable = 0;
78 
79  /* HRTF filterbank coefficients */
80  pData->itds_s = NULL;
81  pData->hrtf_fb = NULL;
82  pData->hrtf_fb_mag = NULL;
83 
84  /* flags/status */
85  pData->progressBar0_1 = 0.0f;
86  pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
87  strcpy(pData->progressBarText,"");
88  pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
89  pData->procStatus = PROC_STATUS_NOT_ONGOING;
90  pData->reInitHRTFsAndGainTables = 1;
91  for(ch=0; ch<MAX_NUM_INPUTS; ch++)
92  pData->recalc_hrtf_interpFLAG[ch] = 1;
93  pData->recalc_M_rotFLAG = 1;
94 }

96 void binauraliser_destroy
97 (
98  void ** const phBin
99 )
100 {
101  binauraliser_data *pData = (binauraliser_data*)(*phBin);
102 
103  if (pData != NULL) {
104  /* not safe to free memory during intialisation/processing loop */
105  while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
106  pData->procStatus == PROC_STATUS_ONGOING){
107  SAF_SLEEP(10);
108  }
109 
110  /* free afSTFT and buffers */
111  if(pData->hSTFT !=NULL)
112  afSTFT_destroy(&(pData->hSTFT));
113  free(pData->inputFrameTD);
114  free(pData->outframeTD);
115  free(pData->inputframeTF);
116  free(pData->outputframeTF);
117  free(pData->hrtf_vbap_gtableComp);
118  free(pData->hrtf_vbap_gtableIdx);
119  free(pData->hrtf_fb);
120  free(pData->hrtf_fb_mag);
121  free(pData->itds_s);
122  free(pData->hrirs);
123  free(pData->hrir_dirs_deg);
124  free(pData->weights);
125  free(pData->progressBarText);
126 
127  free(pData);
128  pData = NULL;
129  }

152 void binauraliser_initCodec
153 (
154  void* const hBin
155 )
156 {
157  binauraliser_data *pData = (binauraliser_data*)(hBin);
158 
159  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
160  return; /* re-init not required, or already happening */
161  while (pData->procStatus == PROC_STATUS_ONGOING){
162  /* re-init required, but we need to wait for the current processing loop to end */
163  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
164  SAF_SLEEP(10);
165  }
166 
167  /* for progress bar */
168  pData->codecStatus = CODEC_STATUS_INITIALISING;
169  strcpy(pData->progressBarText,"Initialising");
170  pData->progressBar0_1 = 0.0f;
171 
172  /* check if TFT needs to be reinitialised */
173  binauraliser_initTFT(hBin);
174 
175  /* reinit HRTFs and interpolation tables */
176  if(pData->reInitHRTFsAndGainTables){
177  binauraliser_initHRTFsAndGainTables(hBin);
178  pData->reInitHRTFsAndGainTables = 0;
179  }
180 
181  /* done! */
182  strcpy(pData->progressBarText,"Done!");
183  pData->progressBar0_1 = 1.0f;
184  pData->codecStatus = CODEC_STATUS_INITIALISED;
185 
186 }

188 void binauraliser_process
189 (
190  void * const hBin,
191  const float *const * inputs,
192  float ** const outputs,
193  int nInputs,
194  int nOutputs,
195  int nSamples
196 )
197 {
198  binauraliser_data *pData = (binauraliser_data*)(hBin);
199  int ch, ear, i, band, nSources;
200  float src_dirs[MAX_NUM_INPUTS][2], Rxyz[3][3], hypotxy;
201  int enableRotation;
202 
203  /* copy user parameters to local variables */
204  nSources = pData->nSources;
205  enableRotation = pData->enableRotation;
206  memcpy(src_dirs, pData->src_dirs_deg, MAX_NUM_INPUTS*2*sizeof(float));
207 
208  /* apply binaural panner */
209  if ((nSamples == BINAURALISER_FRAME_SIZE) && (pData->hrtf_fb!=NULL) && (pData->codecStatus==CODEC_STATUS_INITIALISED) ){
210  pData->procStatus = PROC_STATUS_ONGOING;
211 
212  /* Load time-domain data */
213  for(i=0; i < SAF_MIN(nSources,nInputs); i++)
214  utility_svvcopy(inputs[i], BINAURALISER_FRAME_SIZE, pData->inputFrameTD[i]);
215  for(; i<nSources; i++)
216  memset(pData->inputFrameTD[i], 0, BINAURALISER_FRAME_SIZE * sizeof(float));
217 
218  /* Apply time-frequency transform (TFT) */
219  afSTFT_forward_knownDimensions(pData->hSTFT, pData->inputFrameTD, BINAURALISER_FRAME_SIZE, MAX_NUM_INPUTS, TIME_SLOTS, pData->inputframeTF);
220 
221  /* Rotate source directions */
222  if(enableRotation && pData->recalc_M_rotFLAG){
223  yawPitchRoll2Rzyx (pData->yaw, pData->pitch, pData->roll, pData->useRollPitchYawFlag, Rxyz);
224  for(i=0; i<nSources; i++){
225  pData->src_dirs_xyz[i][0] = cosf(DEG2RAD(pData->src_dirs_deg[i][1])) * cosf(DEG2RAD(pData->src_dirs_deg[i][0]));
226  pData->src_dirs_xyz[i][1] = cosf(DEG2RAD(pData->src_dirs_deg[i][1])) * sinf(DEG2RAD(pData->src_dirs_deg[i][0]));
227  pData->src_dirs_xyz[i][2] = sinf(DEG2RAD(pData->src_dirs_deg[i][1]));
228  pData->recalc_hrtf_interpFLAG[i] = 1;
229  }
230  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSources, 3, 3, 1.0f,
231  (float*)(pData->src_dirs_xyz), 3,
232  (float*)Rxyz, 3, 0.0f,
233  (float*)(pData->src_dirs_rot_xyz), 3);
234  for(i=0; i<nSources; i++){
235  hypotxy = sqrtf(powf(pData->src_dirs_rot_xyz[i][0], 2.0f) + powf(pData->src_dirs_rot_xyz[i][1], 2.0f));
236  pData->src_dirs_rot_deg[i][0] = RAD2DEG(atan2f(pData->src_dirs_rot_xyz[i][1], pData->src_dirs_rot_xyz[i][0]));
237  pData->src_dirs_rot_deg[i][1] = RAD2DEG(atan2f(pData->src_dirs_rot_xyz[i][2], hypotxy));
238  }
239  pData->recalc_M_rotFLAG = 0;
240  }
241 
242  /* interpolate hrtfs and apply to each source */
243  memset(FLATTEN3D(pData->outputframeTF), 0, HYBRID_BANDS*NUM_EARS*TIME_SLOTS * sizeof(float_complex));
244  for (ch = 0; ch < nSources; ch++) {
245  if(pData->recalc_hrtf_interpFLAG[ch]){
246  if(enableRotation)
247  binauraliser_interpHRTFs(hBin, pData->interpMode, pData->src_dirs_rot_deg[ch][0], pData->src_dirs_rot_deg[ch][1], pData->hrtf_interp[ch]);
248  else
249  binauraliser_interpHRTFs(hBin, pData->interpMode, pData->src_dirs_deg[ch][0], pData->src_dirs_deg[ch][1], pData->hrtf_interp[ch]);
250  pData->recalc_hrtf_interpFLAG[ch] = 0;
251  }
252 
253  /* Convolve this channel with the interpolated HRTF, and add it to the binaural buffer */
254  for (band = 0; band < HYBRID_BANDS; band++)
255  for (ear = 0; ear < NUM_EARS; ear++)
256  cblas_caxpy(TIME_SLOTS, &pData->hrtf_interp[ch][band][ear], pData->inputframeTF[band][ch], 1, pData->outputframeTF[band][ear], 1);
257  }
258 
259  /* scale by number of sources */
260  cblas_sscal(/*re+im*/2*HYBRID_BANDS*NUM_EARS*TIME_SLOTS, 1.0f/sqrtf((float)nSources), (float*)FLATTEN3D(pData->outputframeTF), 1);
261 
262  /* inverse-TFT */
263  afSTFT_backward_knownDimensions(pData->hSTFT, pData->outputframeTF, BINAURALISER_FRAME_SIZE, NUM_EARS, TIME_SLOTS, pData->outframeTD);
264 
265  /* Copy to output buffer */
266  for (ch = 0; ch < SAF_MIN(NUM_EARS, nOutputs); ch++)
267  utility_svvcopy(pData->outframeTD[ch], BINAURALISER_FRAME_SIZE, outputs[ch]);
268  for (; ch < nOutputs; ch++)
269  memset(outputs[ch], 0, BINAURALISER_FRAME_SIZE*sizeof(float));
270  }
271  else{
272  for (ch=0; ch < nOutputs; ch++)
273  memset(outputs[ch],0, BINAURALISER_FRAME_SIZE*sizeof(float));
274  }
275 
276  pData->procStatus = PROC_STATUS_NOT_ONGOING;
277 }
