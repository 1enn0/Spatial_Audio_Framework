93 void multiconv_process
94 (
95  void * const hMCnv,
96  const float *const * inputs,
97  float ** const outputs,
98  int nInputs,
99  int nOutputs,
100  int nSamples
101 )
102 {
103  multiconv_data *pData = (multiconv_data*)(hMCnv);
104  int s, ch, i;
105  int numChannels;
106 
107  multiconv_checkReInit(hMCnv);
108 
109  /* prep */
110  numChannels = pData->nChannels;
111 
112  for(s=0; s<nSamples; s++){
113  /* Load input signals into inFIFO buffer */
114  for(ch=0; ch<SAF_MIN(SAF_MIN(nInputs,numChannels),MAX_NUM_CHANNELS); ch++)
115  pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
116  for(; ch<numChannels; ch++) /* Zero any channels that were not given */
117  pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
118 
119  /* Pull output signals from outFIFO buffer */
120  for(ch=0; ch<SAF_MIN(SAF_MIN(nOutputs, numChannels),MAX_NUM_CHANNELS); ch++)
121  outputs[ch][s] = pData->outFIFO[ch][pData->FIFO_idx];
122  for(; ch<nOutputs; ch++) /* Zero any extra channels */
123  outputs[ch][s] = 0.0f;
124 
125  /* Increment buffer index */
126  pData->FIFO_idx++;
127 
128  /* Process frame if inFIFO is full and filters are loaded and saf_matrixConv_apply is ready for it */
129  if (pData->FIFO_idx >= pData->hostBlockSize_clamped && pData->reInitFilters == 0 ) {
130  pData->FIFO_idx = 0;
131 
132  /* Load time-domain data */
133  for(i=0; i < numChannels; i++)
134  utility_svvcopy(pData->inFIFO[i], pData->hostBlockSize_clamped, pData->inputFrameTD[i]);
135 
136  /* Apply convolution */
137  if(pData->hMultiConv != NULL)
138  saf_multiConv_apply(pData->hMultiConv, FLATTEN2D(pData->inputFrameTD), FLATTEN2D(pData->outputFrameTD));
139  else
140  memset(FLATTEN2D(pData->outputFrameTD), 0, MAX_NUM_CHANNELS * (pData->hostBlockSize_clamped)*sizeof(float));
141 
142  /* copy signals to output buffer */
143  for (i = 0; i < SAF_MIN(numChannels, MAX_NUM_CHANNELS); i++)
144  utility_svvcopy(pData->outputFrameTD[i], pData->hostBlockSize_clamped, pData->outFIFO[i]);
145  }
146  else if(pData->FIFO_idx >= pData->hostBlockSize_clamped){
147  /* clear outFIFO if codec was not ready */
148  pData->FIFO_idx = 0;
149  memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
150  }
151  }
152 }

163 void multiconv_checkReInit(void* const hMCnv)
164 {
165  multiconv_data *pData = (multiconv_data*)(hMCnv);
166 
167  /* reinitialise if needed */
168  if ((pData->reInitFilters == 1) && (pData->filters !=NULL)) {
169  pData->reInitFilters = 2;
170  saf_multiConv_destroy(&(pData->hMultiConv));
171  pData->hostBlockSize_clamped = SAF_CLAMP(pData->hostBlockSize, MIN_FRAME_SIZE, MAX_FRAME_SIZE);
172  saf_multiConv_create(&(pData->hMultiConv),
173  pData->hostBlockSize_clamped,
174  pData->filters,
175  pData->filter_length,
176  pData->nfilters,
177  pData->enablePartitionedConv);
178 
179  /* Resize buffers */
180  pData->inputFrameTD = (float**)realloc2d((void**)pData->inputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
181  pData->outputFrameTD = (float**)realloc2d((void**)pData->outputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
182  memset(FLATTEN2D(pData->inputFrameTD), 0, MAX_NUM_CHANNELS*(pData->hostBlockSize_clamped)*sizeof(float));
183 
184  /* reset FIFO buffers */
185  pData->FIFO_idx = 0;
186  memset(pData->inFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
187  memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
188 
189  pData->reInitFilters = 0;
190  }
191 }
