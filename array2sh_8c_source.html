52 void array2sh_create
53 (
54  void ** const phA2sh
55 )
56 {
57  array2sh_data* pData = (array2sh_data*)malloc1d(sizeof(array2sh_data));
58  *phA2sh = (void*)pData;
59 
60  /* defualt parameters */
61  array2sh_createArray(&(pData->arraySpecs));
62  pData->filterType = FILTER_TIKHONOV;
63  pData->regPar = 15.0f;
64  pData->chOrdering = CH_ACN;
65  pData->norm = NORM_SN3D;
66  pData->c = 343.0f;
67  pData->gain_dB = 0.0f; /* post-gain */
68  array2sh_arrayPars* arraySpecs = (array2sh_arrayPars*)(pData->arraySpecs);
69  array2sh_initArray(arraySpecs, MICROPHONE_ARRAY_PRESET_DEFAULT, &(pData->order), 1);
70  pData->enableDiffEQpastAliasing = 1;
71 
72  /* time-frequency transform + buffers */
73  pData->hSTFT = NULL;
74  pData->inputFrameTD = (float**)malloc2d(MAX_NUM_SENSORS, ARRAY2SH_FRAME_SIZE, sizeof(float));
75  pData->SHframeTD = (float**)malloc2d(MAX_NUM_SH_SIGNALS, ARRAY2SH_FRAME_SIZE, sizeof(float));
76  pData->inputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SENSORS, TIME_SLOTS, sizeof(float_complex));
77  pData->SHframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
78 
79  /* internal */
80  pData->progressBar0_1 = 0.0f;
81  pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
82  strcpy(pData->progressBarText,"");
83  pData->evalStatus = EVAL_STATUS_NOT_EVALUATED;
84  pData->evalRequestedFLAG = 0;
85  pData->reinitSHTmatrixFLAG = 1;
86  pData->new_order = pData->order;
87  pData->bN = NULL;
88 
89  /* display related stuff */
90  pData->bN_modal_dB = (float**)calloc2d(HYBRID_BANDS, MAX_SH_ORDER + 1, sizeof(float));
91  pData->bN_inv_dB = (float**)calloc2d(HYBRID_BANDS, MAX_SH_ORDER + 1, sizeof(float));
92  pData->cSH = (float*)calloc1d((HYBRID_BANDS)*(MAX_SH_ORDER + 1),sizeof(float));
93  pData->lSH = (float*)calloc1d((HYBRID_BANDS)*(MAX_SH_ORDER + 1),sizeof(float));
94 }

96 void array2sh_destroy
97 (
98  void ** const phM2sh
99 )
100 {
101  array2sh_data *pData = (array2sh_data*)(*phM2sh);
102 
103  if (pData != NULL) {
104  /* not safe to free memory during evaluation */
105  while (pData->evalStatus == EVAL_STATUS_EVALUATING)
106  SAF_SLEEP(10);
107 
108  /* free afSTFT and buffers */
109  if (pData->hSTFT != NULL)
110  afSTFT_destroy(&(pData->hSTFT));
111  free(pData->inputFrameTD);
112  free(pData->SHframeTD);
113  free(pData->inputframeTF);
114  free(pData->SHframeTF);
115  array2sh_destroyArray(&(pData->arraySpecs));
116 
117  /* Display stuff */
118  free(pData->bN_modal_dB);
119  free(pData->bN_inv_dB);
120 
121  free(pData->progressBarText);
122 
123  free(pData->bN);
124  free(pData->cSH);
125  free(pData->lSH);
126 
127  free(pData);
128  pData = NULL;
129  }

147  void* const hA2sh
148 )
149 {
150  array2sh_data *pData = (array2sh_data*)(hA2sh);
151 
152  if (pData->evalStatus != EVAL_STATUS_NOT_EVALUATED)

169 void array2sh_process
170 (
171  void * const hA2sh,
172  const float *const * inputs,
173  float ** const outputs,
174  int nInputs,
175  int nOutputs,
176  int nSamples
177 )
178 {
179  array2sh_data *pData = (array2sh_data*)(hA2sh);
180  array2sh_arrayPars* arraySpecs = (array2sh_arrayPars*)(pData->arraySpecs);
181  int ch, i, band, Q, order, nSH;
182  const float_complex calpha = cmplxf(1.0f,0.0f), cbeta = cmplxf(0.0f, 0.0f);
183  CH_ORDER chOrdering;
184  NORM_TYPES norm;
185  float gain_lin;
186 
187  /* reinit TFT if needed */
188  array2sh_initTFT(hA2sh);
189 
190  /* compute encoding matrix if needed */
191  if (pData->reinitSHTmatrixFLAG) {
192  array2sh_calculate_sht_matrix(hA2sh); /* compute encoding matrix */
193  array2sh_calculate_mag_curves(hA2sh); /* calculate magnitude response curves */
194  pData->reinitSHTmatrixFLAG = 0;
195  }
196 
197  /* local copy of user parameters */
198  chOrdering = pData->chOrdering;
199  norm = pData->norm;
200  gain_lin = powf(10.0f, pData->gain_dB/20.0f);
201  Q = arraySpecs->Q;
202  order = pData->order;
203  nSH = (order+1)*(order+1);
204 
205  /* processing loop */
206  if ((nSamples == ARRAY2SH_FRAME_SIZE) && (pData->reinitSHTmatrixFLAG==0) ) {
207  pData->procStatus = PROC_STATUS_ONGOING;
208 
209  /* Load time-domain data */
210  for(i=0; i < nInputs; i++)
211  utility_svvcopy(inputs[i], ARRAY2SH_FRAME_SIZE, pData->inputFrameTD[i]);
212  for(; i<Q; i++)
213  memset(pData->inputFrameTD[i], 0, ARRAY2SH_FRAME_SIZE * sizeof(float));
214 
215  /* Apply time-frequency transform (TFT) */
216  afSTFT_forward_knownDimensions(pData->hSTFT, pData->inputFrameTD, ARRAY2SH_FRAME_SIZE, MAX_NUM_SENSORS, TIME_SLOTS, pData->inputframeTF);
217 
218  /* Apply spherical harmonic transform (SHT) */
219  for(band=0; band<HYBRID_BANDS; band++){
220  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSH, TIME_SLOTS, Q, &calpha,
221  pData->W[band], MAX_NUM_SENSORS,
222  FLATTEN2D(pData->inputframeTF[band]), TIME_SLOTS, &cbeta,
223  FLATTEN2D(pData->SHframeTF[band]), TIME_SLOTS);
224  }
225 
226  /* inverse-TFT */
227  afSTFT_backward_knownDimensions(pData->hSTFT, pData->SHframeTF, ARRAY2SH_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->SHframeTD);
228 
229  /* account for output channel order */
230  switch(chOrdering){
231  case CH_ACN: /* already ACN, do nothing */ break;
232  case CH_FUMA: convertHOAChannelConvention(FLATTEN2D(pData->SHframeTD), order, ARRAY2SH_FRAME_SIZE, HOA_CH_ORDER_ACN, HOA_CH_ORDER_FUMA); break;
233  }
234 
235  /* account for normalisation scheme */
236  switch(norm){
237  case NORM_N3D: /* already N3D, do nothing */ break;
238  case NORM_SN3D: convertHOANormConvention(FLATTEN2D(pData->SHframeTD), order, ARRAY2SH_FRAME_SIZE, HOA_NORM_N3D, HOA_NORM_SN3D); break;
239  case NORM_FUMA: convertHOANormConvention(FLATTEN2D(pData->SHframeTD), order, ARRAY2SH_FRAME_SIZE, HOA_NORM_N3D, HOA_NORM_FUMA); break;
240  }
241 
242  /* Apply post-gain */
243  utility_svsmul(FLATTEN2D(pData->SHframeTD), &gain_lin, nSH*ARRAY2SH_FRAME_SIZE, NULL);
244 
245  /* Copy to output */
246  for(i = 0; i < SAF_MIN(nSH,nOutputs); i++)
247  utility_svvcopy(pData->SHframeTD[i], ARRAY2SH_FRAME_SIZE, outputs[i]);
248  for(; i < nOutputs; i++)
249  memset(outputs[i], 0, ARRAY2SH_FRAME_SIZE * sizeof(float));
250  }
251  else{
252  for (ch=0; ch < nOutputs; ch++)
253  memset(outputs[ch],0, ARRAY2SH_FRAME_SIZE*sizeof(float));
254  }
255 
256  pData->procStatus = PROC_STATUS_NOT_ONGOING;
257 }

268 void array2sh_setEncodingOrder(void* const hA2sh, int newOrder)
269 {
270  array2sh_data *pData = (array2sh_data*)(hA2sh);
271 
272  if(pData->new_order != newOrder){
273  pData->new_order = newOrder;
274  pData->reinitSHTmatrixFLAG = 1;
275  array2sh_setEvalStatus(hA2sh, EVAL_STATUS_NOT_EVALUATED);
276  }
277  /* FUMA only supports 1st order */
278  if(pData->new_order!=SH_ORDER_FIRST && pData->chOrdering == CH_FUMA)
279  pData->chOrdering = CH_ACN;
280  if(pData->new_order!=SH_ORDER_FIRST && pData->norm == NORM_FUMA)
281  pData->norm = NORM_SN3D;
