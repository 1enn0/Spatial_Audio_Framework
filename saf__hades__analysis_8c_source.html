43void hades_analysis_create
44(
45 hades_analysis_handle* const phAna,
46 float fs,
47 HADES_FILTERBANKS fbOption,
48 int hopsize,
49 int blocksize,
50 int hybridmode,
51 float* h_array,
52 float* grid_dirs_deg,
53 int nGrid,
54 int nMics,
55 int h_len,
56 HADES_DIFFUSENESS_ESTIMATORS diffOption,
57 HADES_DOA_ESTIMATORS doaOption
58)
59{
60 hades_analysis_data* a = (hades_analysis_data*)malloc1d(sizeof(hades_analysis_data));
61 *phAna = (void*)a;
62 int band, i, idx_max;
63 float* w_tmp;
64 float_complex *U, *E, *H_W;
65 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f); /* blas */
66
67 nMics = SAF_MIN(nMics, HADES_MAX_NMICS);
68 assert(blocksize % hopsize == 0); /* Must be a multiple of hopsize */
69 assert(blocksize<=HADES_MAX_BLOCKSIZE);
70
71 /* User parameters */
72 a->fs = fs;
73 a->fbOpt = fbOption;
74 a->hopsize = hopsize;
75 a->blocksize = blocksize;
76 a->hybridmode = hybridmode;
77 a->h_array = malloc1d(nGrid*nMics*h_len*sizeof(float));
78 memcpy(a->h_array, h_array, nGrid*nMics*h_len*sizeof(float));
79 a->grid_dirs_deg = malloc1d(nGrid*2*sizeof(float));
80 memcpy(a->grid_dirs_deg, grid_dirs_deg, nGrid*2*sizeof(float));
81 a->nGrid = nGrid;
82 a->nMics = nMics;
83 a->h_len = h_len;
84 a->diffOpt = diffOption;
85 a->doaOpt = doaOption;
86 a->covAvgCoeff = 1.0f - 1.0f/(4096.0f/a->blocksize);
87 a->covAvgCoeff = SAF_CLAMP(a->covAvgCoeff, 0.0f, 0.99999f);
88
89 /* Scale steering vectors so that the peak of loudest measurement is 1 */
90 utility_simaxv(a->h_array, nGrid*nMics*h_len, &idx_max);
91 cblas_sscal(nGrid*nMics*h_len, 1.0f/a->h_array[idx_max], a->h_array, 1);
92
93 /* Initialise time-frequency transform */
94 a->timeSlots = a->blocksize/a->hopsize;
95 switch(a->fbOpt){
96 case HADES_USE_AFSTFT_LD: /* fall through */
97 case HADES_USE_AFSTFT:
98 afSTFT_create(&(a->hFB_enc), a->nMics, 0, a->hopsize, a->fbOpt==HADES_USE_AFSTFT_LD ? 1 : 0, a->hybridmode, AFSTFT_BANDS_CH_TIME);
99 a->nBands = afSTFT_getNBands(a->hFB_enc);
100 a->freqVector = malloc1d(a->nBands*sizeof(float));
101 a->filterbankDelay = afSTFT_getProcDelay(a->hFB_enc);
102 afSTFT_getCentreFreqs(a->hFB_enc, a->fs, a->nBands, a->freqVector);
103 a->H_array = malloc1d(a->nBands*(a->nMics)*(a->nGrid)*sizeof(float_complex));
104 a->H_array_w = malloc1d(a->nBands*(a->nMics)*(a->nGrid)*sizeof(float_complex));
105 afSTFT_FIRtoFilterbankCoeffs(a->h_array, a->nGrid, a->nMics, a->h_len, a->hopsize, a->fbOpt==HADES_USE_AFSTFT_LD ? 1 : 0, a->hybridmode, a->H_array);
106 break;
107 }
108
109 /* Initialise DoA estimator */
110 utility_cseig_create(&(a->hEig), a->nMics);
111 a->grid_dirs_xyz = malloc1d(a->nGrid*3*sizeof(float));
112 unitSph2cart(a->grid_dirs_deg, a->nGrid, 1, a->grid_dirs_xyz);
113 switch(a->doaOpt){
114 case HADES_USE_MUSIC: hades_sdMUSIC_create(&(a->hDoA), a->nMics, a->grid_dirs_deg, a->nGrid); break;
115 }
116
117 /* Integration weights */
118 a->W = calloc1d(a->nGrid*a->nGrid,sizeof(float_complex));
119 if (cblas_sasum(a->nGrid, a->grid_dirs_deg+1, 2)/(float)a->nGrid<0.0001){
120 for(i=0; i<a->nGrid; i++)
121 a->W[i*(a->nGrid)+i] = cmplxf(1.0f, 0.0f);
122 }
123 else{
124 w_tmp = malloc1d(a->nGrid*sizeof(float));
125 getVoronoiWeights(a->grid_dirs_deg, a->nGrid, 0, w_tmp);
126 for(i=0; i<a->nGrid; i++)
127 a->W[i*(a->nGrid)+i] = cmplxf(w_tmp[i], 0.0f);
128 }
129
130 /* For spatial whitening of the spatial covariance matrix, such that it has an identity structure under diffuse-field conditions */
131 a->T = (float_complex**)malloc2d(a->nBands, a->nMics*(a->nMics), sizeof(float_complex));
132 a->DCM_array = malloc1d(a->nBands*(a->nMics)*(a->nMics)*sizeof(float_complex));
133 U = malloc1d(a->nMics*(a->nMics)*sizeof(float_complex));
134 E = malloc1d(a->nMics*(a->nMics)*sizeof(float_complex));
135 H_W = malloc1d(a->nMics*(a->nGrid)*sizeof(float_complex));
136 for(band=0; band<a->nBands; band++){
137 /* Diffuse covariance matrix */
138 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, a->nMics, a->nGrid, a->nGrid, &calpha,
139 &(a->H_array[band*(a->nMics)*(a->nGrid)]), a->nGrid,
140 a->W, a->nGrid, &cbeta,
141 H_W, a->nGrid);
142 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, a->nMics, a->nMics, a->nGrid, &calpha,
143 H_W, a->nGrid,
144 &(a->H_array[band*(a->nMics)*(a->nGrid)]), a->nGrid, &cbeta,
145 &(a->DCM_array[band*(a->nMics)*(a->nMics)]), a->nMics);
146 cblas_sscal(/*re+im*/2*(a->nMics)*(a->nMics), 1.0f/(float)a->nGrid, (float*)&(a->DCM_array[band*(a->nMics)*(a->nMics)]), 1);
147
148 /* Decomposition of the diffuse covariance matrix */
149 utility_cseig(a->hEig, &(a->DCM_array[band*(a->nMics)*(a->nMics)]), a->nMics, 1, U, E, NULL);
150
151 /* Compute spatial whitening matrix */
152 for(i=0; i<a->nMics; i++)
153 E[i*a->nMics+i] = cmplxf(sqrtf(1.0f/(crealf(E[i*a->nMics+i])+2.23e-10f)), 0.0f);
154 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, a->nMics, a->nMics, a->nMics, &calpha,
155 E, a->nMics,
156 U, a->nMics, &cbeta,
157 a->T[band], a->nMics);
158
159 /* Whiten the array steering vectors / anechoic relative transfer functions (RTFs) */
160 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, a->nMics, a->nGrid, a->nMics, &calpha,
161 a->T[band], a->nMics,
162 &(a->H_array[band*(a->nMics)*(a->nGrid)]), a->nGrid, &cbeta,
163 &(a->H_array_w[band*(a->nMics)*(a->nGrid)]), a->nGrid);
164 }
165 free(U);
166 free(E);
167 free(H_W);
168
169 /* Run-time variables */
170 a->inputBlock = (float**)malloc2d(a->nMics, a->blocksize, sizeof(float));
171 a->Cx = malloc1d(a->nBands*sizeof(CxMic));
172 a->V = malloc1d((a->nMics)*(a->nMics)*sizeof(float_complex));
173 a->Vn = malloc1d((a->nMics)*(a->nMics)*sizeof(float_complex));
174 a->lambda = malloc1d((a->nMics)*sizeof(float));
175
176 /* Flush run-time buffers with zeros */
177 hades_analysis_reset((*phAna));
178}

180void hades_analysis_destroy
181(
182 hades_analysis_handle* const phAna
183)
184{
185 hades_analysis_data *a = (hades_analysis_data*)(*phAna);
186
187 if (a != NULL) {
188 free(a->h_array);
189 free(a->H_array);
190 free(a->H_array_w);
191 free(a->DCM_array);
192 free(a->W);
193 free(a->T);
194 free(a->grid_dirs_xyz);
195 free(a->grid_dirs_deg);
196
197 /* Destroy time-frequency transform */
198 switch(a->fbOpt){
199 case HADES_USE_AFSTFT_LD: /* fall through */
200 case HADES_USE_AFSTFT: afSTFT_destroy(&(a->hFB_enc)); break;
201 }
202 free(a->freqVector);
203
204 /* Destroy DoA estimator */
205 utility_cseig_destroy(&(a->hEig));
206 switch(a->doaOpt){
207 case HADES_USE_MUSIC:
208 hades_sdMUSIC_destroy(&(a->hDoA));
209 break;
210 }
211
212 /* Free run-time variables */
213 free(a->inputBlock);
214 free(a->Cx);
215 free(a->V);
216 free(a->Vn);
217 free(a->lambda);
218
219 free(a);
220 a = NULL;
221 (*phAna) = NULL;
222 }

240void hades_analysis_apply
241(
242 hades_analysis_handle const hAna,
243 float** input,
244 int nChannels,
245 int blocksize,
246 void* const hPCon,
247 void* const hSCon
248)
249{
250 hades_analysis_data *a = (hades_analysis_data*)(hAna);
251 hades_param_container_data *pcon = (hades_param_container_data*)(hPCon);
252 hades_signal_container_data *scon = (hades_signal_container_data*)(hSCon);
253 int i, j, k, ch, band, est_idx;
254 float diffuseness;
255 CxMic Cx_new, T_Cx, T_Cx_TH;
256 const float_complex calpha = cmplxf(1.0f, 0.0f); const float_complex cbeta = cmplxf(0.0f, 0.0f); /* blas */
257
258 assert(blocksize==a->blocksize);
259
260 /* Load time-domain data */
261 for(ch=0; ch<SAF_MIN(nChannels, a->nMics); ch++)
262 cblas_scopy(blocksize, input[ch], 1, a->inputBlock[ch], 1);
263 for(; ch<a->nMics; ch++)
264 memset(a->inputBlock[ch], 0, blocksize*sizeof(float));
265
266 /* Forward time-frequency transform */
267 switch(a->fbOpt){
268 case HADES_USE_AFSTFT_LD: /* fall through */
269 case HADES_USE_AFSTFT: afSTFT_forward_knownDimensions(a->hFB_enc, a->inputBlock, blocksize, a->nMics, a->timeSlots, scon->inTF); break;
270 }
271
272 /* Update covarience matrix per band */
273 for(band=0; band<a->nBands; band++){
274 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, a->nMics, a->nMics, a->timeSlots, &calpha,
275 FLATTEN2D(scon->inTF[band]), a->timeSlots,
276 FLATTEN2D(scon->inTF[band]), a->timeSlots, &cbeta,
277 Cx_new.Cx, a->nMics);
278
279 /* Make a copy for the signal container */
280 cblas_ccopy(a->nMics*a->nMics, (float_complex*)Cx_new.Cx, 1, (float_complex*)scon->Cx[band].Cx, 1);
281
282 /* Apply temporal averaging */
283 cblas_sscal(/*re+im*/2*(a->nMics) * (a->nMics), SAF_CLAMP(a->covAvgCoeff, 0.0f, 0.999f), (float*)a->Cx[band].Cx, 1);
284 cblas_saxpy(/*re+im*/2*(a->nMics) * (a->nMics), 1.0f-SAF_CLAMP(a->covAvgCoeff, 0.0f, 0.999f), (float*)Cx_new.Cx, 1, (float*)a->Cx[band].Cx, 1);
285 }
286
287 /* Spatial parameter estimation per band */
288 for (band = 0; band < a->nBands; band++) {
289 /* Apply diffuse whitening process */
290 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, a->nMics, a->nMics, a->nMics, &calpha,
291 a->T[band], a->nMics,
292 a->Cx[band].Cx, a->nMics, &cbeta,
293 T_Cx.Cx, a->nMics);
294 cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, a->nMics, a->nMics, a->nMics, &calpha,
295 T_Cx.Cx, a->nMics,
296 a->T[band], a->nMics, &cbeta,
297 T_Cx_TH.Cx, a->nMics);
298 utility_cseig(a->hEig, T_Cx_TH.Cx, a->nMics, 1, a->V, NULL, a->lambda);
299
300 /* Estimate diffuseness */
301 switch(a->diffOpt){
302 case HADES_USE_COMEDIE: diffuseness = hades_comedie(a->lambda, a->nMics); break;
303 }
304
305 /* Store diffuseness and source number estimates */
306 pcon->diffuseness[band] = diffuseness;
307 pcon->gains_dir[band] = pcon->gains_diff[band] = 1.0f; /* Default gains per band */
308
309 /* Apply DoA estimator */
310 switch(a->doaOpt){
311 case HADES_USE_MUSIC:
312 /* perform sphMUSIC on the noise subspace */
313 for(i=0; i<a->nMics; i++)
314 for(j=0, k=1; j<a->nMics-1; j++, k++)
315 a->Vn[i*(a->nMics-1)+j] = a->V[i*(a->nMics)+k];
316 hades_sdMUSIC_compute(a->hDoA, &(a->H_array_w[band*(a->nMics)*(a->nGrid)]), a->Vn, 1, NULL, &est_idx);
317 break;
318 }
319
320 /* Store */
321 pcon->doa_idx[band] = pcon->gains_idx[band] = est_idx;
322 }
323}
