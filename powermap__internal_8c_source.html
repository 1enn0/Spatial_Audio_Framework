45 void powermap_initAna(void* const hPm)
46 {
47  powermap_data *pData = (powermap_data*)(hPm);
48  powermap_codecPars* pars = pData->pars;
49  int i, j, n, N_azi, N_ele, nSH_order, order;
50  float scaleY, hfov, vfov, fi, aspectRatio;
51  float* Y_grid_N, *grid_x_axis, *grid_y_axis;
52 
53  order = pData->new_masterOrder;
54 
55  /* Store Y_grid per order */
56  int geosphere_ico_freq = 9;
57  pars->grid_dirs_deg = (float*)__HANDLES_geosphere_ico_dirs_deg[geosphere_ico_freq];
58  pars->grid_nDirs = __geosphere_ico_nPoints[geosphere_ico_freq];
59  Y_grid_N = malloc1d(((order+1)*(order+1))*(pars->grid_nDirs)*sizeof(float));
60  getRSH(order, pars->grid_dirs_deg, pars->grid_nDirs, Y_grid_N);
61  for(n=1; n<=order; n++){
62  nSH_order = (n+1)*(n+1);
63  scaleY = 1.0f/(float)nSH_order;
64  free(pars->Y_grid[n-1]);
65  free(pars->Y_grid_cmplx[n-1]);
66  pars->Y_grid[n-1] = malloc1d(nSH_order * (pars->grid_nDirs)*sizeof(float));
67  pars->Y_grid_cmplx[n-1] = malloc1d(nSH_order * (pars->grid_nDirs)*sizeof(float_complex));
68  memcpy(pars->Y_grid[n-1], Y_grid_N, nSH_order * (pars->grid_nDirs)*sizeof(float));
69  utility_svsmul(pars->Y_grid[n-1], &scaleY, nSH_order * (pars->grid_nDirs), NULL);
70  for(i=0; i<nSH_order; i++)
71  for(j=0; j<pars->grid_nDirs; j++)
72  pars->Y_grid_cmplx[n-1][i*(pars->grid_nDirs)+j] = cmplxf(pars->Y_grid[n-1][i*(pars->grid_nDirs)+j], 0.0f);
73  }
74 
75  /* generate interpolation table for current display settings */
76  switch(pData->HFOVoption){
77  default:
78  case HFOV_360: hfov = 360.0f; break;
79  }
80  switch(pData->aspectRatioOption){
81  default:
82  case ASPECT_RATIO_2_1: aspectRatio = 2.0f; break;
83  }
84  N_azi = pData->dispWidth;
85  N_ele = (int)((float)pData->dispWidth/aspectRatio + 0.5f);
86  grid_x_axis = malloc1d(N_azi * sizeof(float));
87  grid_y_axis = malloc1d(N_ele * sizeof(float));
88  vfov = hfov/aspectRatio;
89  for(fi = -hfov/2.0f, i = 0; i<N_azi; fi+=hfov/N_azi, i++)
90  grid_x_axis[i] = fi;
91  for(fi = -vfov/2.0f, i = 0; i<N_ele; fi+=vfov/N_ele, i++)
92  grid_y_axis[i] = fi;
93  free(pars->interp_dirs_deg);
94  pars->interp_dirs_deg = malloc1d(N_azi*N_ele*2*sizeof(float));
95  for(i = 0; i<N_ele; i++){
96  for(j=0; j<N_azi; j++){
97  pars->interp_dirs_deg[(i*N_azi + j)*2] = grid_x_axis[j];
98  pars->interp_dirs_deg[(i*N_azi + j)*2+1] = grid_y_axis[i];
99  }
100  }
101  free(pars->interp_table);
102  generateVBAPgainTable3D_srcs(pars->interp_dirs_deg, N_azi*N_ele, pars->grid_dirs_deg, pars->grid_nDirs, 0, 0, 0.0f, &(pars->interp_table), &(pars->interp_nDirs), &(pars->interp_nTri));
103  VBAPgainTable2InterpTable(pars->interp_table, pars->interp_nDirs, pars->grid_nDirs);
104 
105  /* reallocate memory for storing the powermaps */
106  free(pData->pmap);
107  pData->pmap = malloc1d(pars->grid_nDirs*sizeof(float));
108  free(pData->prev_pmap);
109  pData->prev_pmap = calloc1d(pars->grid_nDirs, sizeof(float));
110  for(i=0; i<NUM_DISP_SLOTS; i++){
111  free(pData->pmap_grid[i]);
112  pData->pmap_grid[i] = calloc1d(pars->interp_nDirs,sizeof(float));
113  }
114 
115  pData->masterOrder = order;
116 
117  free(Y_grid_N);
118  free(grid_x_axis);
119  free(grid_y_axis);
120 }
