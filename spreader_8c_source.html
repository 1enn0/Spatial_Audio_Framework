27 void spreader_create

35

64  pData->weights = NULL;

67  pData->hDecor[src] = NULL;
68  pData->Cy[src] = NULL;
69  pData->Cproto[src] = NULL;
70  pData->prev_M[src] = NULL;
71  pData->prev_Mr[src] = NULL;
72  pData->dirActive[src] = NULL;

75  pData->new_Mr = NULL;

178 void spreader_initCodec
179 (
180  void* const hSpr
181 )
182 {
183  spreader_data *pData = (spreader_data*)(hSpr);
184  int q, band, ng, nSources, src;
185  float_complex scaleC;
186 #ifdef SAF_ENABLE_SOFA_READER_MODULE
187  saf_sofa_container sofa;
188  SAF_SOFA_ERROR_CODES error;
189 #endif
190  SPREADER_PROC_MODES procMode;
191  float_complex H_tmp[MAX_NUM_CHANNELS];
192  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
193 
194  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
195  return; /* re-init not required, or already happening */
196  while (pData->procStatus == PROC_STATUS_ONGOING){
197  /* re-init required, but we need to wait for the current processing loop to end */
198  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
199  SAF_SLEEP(10);
200  }
201 
202  nSources = pData->new_nSources;
203  procMode = pData->new_procMode;
204 
205  /* for progress bar */
206  pData->codecStatus = CODEC_STATUS_INITIALISING;
207  strcpy(pData->progressBarText,"Initialising");
208  pData->progressBar0_1 = 0.0f;
209 
210  /* Load measurements (e.g. HRIRs, Microphone array IRs etc.) */
211 #ifndef SAF_ENABLE_SOFA_READER_MODULE
212  pData->useDefaultHRIRsFLAG = 1;
213 #endif
214  if(pData->useDefaultHRIRsFLAG){
215  /* Load default HRIR data */
216  pData->Q = NUM_EARS;
217  pData->nGrid = __default_N_hrir_dirs;
218  pData->h_len = __default_hrir_len;
219  pData->h_fs = (float)__default_hrir_fs;
220  pData->h_grid = realloc1d(pData->h_grid, pData->nGrid * (pData->Q) * (pData->h_len) * sizeof(float));
221  memcpy(pData->h_grid, (float*)__default_hrirs, pData->nGrid * (pData->Q) * (pData->h_len) * sizeof(float));
222  pData->grid_dirs_deg = realloc1d(pData->grid_dirs_deg, pData->nGrid * 2 * sizeof(float));
223  memcpy(pData->grid_dirs_deg, (float*)__default_hrir_dirs_deg, pData->nGrid * 2 * sizeof(float));
224  }
225 #ifdef SAF_ENABLE_SOFA_READER_MODULE
226  else{
227  /* Use sofa loader */
228  error = saf_sofa_open(&sofa, pData->sofa_filepath);
229  if(error!=SAF_SOFA_OK){
230  /* if failed, then load default data instead */
231  pData->useDefaultHRIRsFLAG = 1;
232  saf_print_warning("Unable to load the specified SOFA file. Using default HRIR data instead");
233  spreader_initCodec(hSpr);
234  }
235  pData->h_fs = sofa.DataSamplingRate;
236  pData->h_len = sofa.DataLengthIR;
237  pData->nGrid = sofa.nSources;
238  pData->h_grid = realloc1d(pData->h_grid, pData->nGrid*(pData->Q)*(pData->h_len)*sizeof(float));
239  memcpy(pData->h_grid, sofa.DataIR, pData->nGrid*(pData->Q)*(pData->h_len)*sizeof(float));
240  pData->grid_dirs_deg = realloc1d(pData->grid_dirs_deg, pData->nGrid*2*sizeof(float));
241  cblas_scopy(pData->nGrid, sofa.SourcePosition, 3, pData->grid_dirs_deg, 2); /* azi */
242  cblas_scopy(pData->nGrid, &sofa.SourcePosition[1], 3, &pData->grid_dirs_deg[1], 2); /* elev */
243  saf_sofa_close(&sofa);
244  }
245 #endif
246 
247  /* Convert from the 0..360 convention, to -180..180, and pre-compute unit Cartesian vectors */
248  convert_0_360To_m180_180(pData->grid_dirs_deg, pData->nGrid);
249  pData->grid_dirs_xyz = realloc1d(pData->grid_dirs_xyz, pData->nGrid*3*sizeof(float));
250  unitSph2cart(pData->grid_dirs_deg, pData->nGrid, 1, pData->grid_dirs_xyz);
251 
252  /* Initialise time-frequency transform and decorrelators */
253  afSTFT_destroy(&(pData->hSTFT));
254  afSTFT_create(&(pData->hSTFT), nSources, pData->Q, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
255  int orders[4] = {20, 15, 6, 6}; /* 20th order up to 700Hz, 15th->2.4kHz, 6th->4kHz, 3rd->12kHz, NONE(only delays)->Nyquist */
256  //float freqCutoffs[4] = {600.0f, 2.6e3f, 4.5e3f, 12e3f};
257  float freqCutoffs[4] = {900.0f, 6.8e3f, 12e3f, 24e3f};
258  const int maxDelay = 12;
259  for(src=0; src<SPREADER_MAX_NUM_SOURCES; src++){
260  latticeDecorrelator_destroy(&(pData->hDecor[src]));
261  latticeDecorrelator_create(&(pData->hDecor[src]), (float)pData->fs, HOP_SIZE, pData->freqVector, HYBRID_BANDS, pData->Q, orders, freqCutoffs, 4, maxDelay, 0, 0.75f);
262  }
263 
264  /* Convert to filterbank coefficients and pre-compute outer products */
265  pData->H_grid = realloc1d(pData->H_grid, HYBRID_BANDS*(pData->Q)*pData->nGrid*sizeof(float_complex));
266  afSTFT_FIRtoFilterbankCoeffs(pData->h_grid, pData->nGrid, pData->Q, pData->h_len, HOP_SIZE, 0, 1, pData->H_grid);
267  pData->weights = realloc1d(pData->weights, pData->nGrid*sizeof(float));
268  getVoronoiWeights(pData->grid_dirs_deg, pData->nGrid, 0, pData->weights);
269  cblas_sscal(pData->nGrid, 1.0f/FOURPI, pData->weights, 1);
270  for(band=0; band<HYBRID_BANDS; band++){
271  pData->HHH[band] = (float_complex**)realloc2d((void**)pData->HHH[band], pData->nGrid, pData->Q * (pData->Q), sizeof(float_complex));
272  for(ng=0; ng<pData->nGrid; ng++){
273  for(q=0; q<pData->Q; q++)
274  H_tmp[q] = pData->H_grid[band*(pData->Q)*pData->nGrid + q*pData->nGrid + ng];
275  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, pData->Q, pData->Q, 1, &calpha,
276  H_tmp, 1,
277  H_tmp, 1, &cbeta,
278  pData->HHH[band][ng], pData->Q);
279  scaleC = cmplxf(pData->weights[ng], 0.0f);
280  cblas_cscal(pData->Q * (pData->Q), &scaleC, pData->HHH[band][ng], 1);
281  }
282  }
283  pData->angles = realloc1d(pData->angles, pData->nGrid*sizeof(float));
284 
285  /* OM structures */
286  cdf4sap_cmplx_destroy(&(pData->hCdf));
287  cdf4sap_cmplx_create(&(pData->hCdf), pData->Q, pData->Q);
288  cdf4sap_destroy(&(pData->hCdf_res));
289  cdf4sap_create(&(pData->hCdf_res), pData->Q, pData->Q);
290  pData->Qmix = realloc1d(pData->Qmix, pData->Q*(pData->Q)*sizeof(float));
291  memset(pData->Qmix, 0, pData->Q*(pData->Q)*sizeof(float));
292  pData->Qmix_cmplx = realloc1d(pData->Qmix_cmplx, pData->Q*(pData->Q)*sizeof(float_complex));
293  memset(pData->Qmix_cmplx, 0, pData->Q*(pData->Q)*sizeof(float_complex));
294  for(q=0; q<pData->Q; q++){
295  pData->Qmix[q*(pData->Q)+q] = 1.0f;
296  pData->Qmix_cmplx[q*(pData->Q)+q] = cmplxf(1.0f, 0.0f);
297  }
298  pData->Cr = realloc1d(pData->Cr, pData->Q*(pData->Q)*sizeof(float));
299  pData->Cr_cmplx = realloc1d(pData->Cr_cmplx, pData->Q*(pData->Q)*sizeof(float_complex));
300 
301  /* mixing matrices and buffers */
302  for(src=0; src<SPREADER_MAX_NUM_SOURCES; src++){
303  pData->Cy[src] = (float_complex**)realloc2d((void**)pData->Cy[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
304  memset(FLATTEN2D(pData->Cy[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float_complex));
305  pData->Cproto[src] = (float_complex**)realloc2d((void**)pData->Cproto[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
306  memset(FLATTEN2D(pData->Cproto[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float_complex));
307  pData->prev_M[src] = (float_complex**)realloc2d((void**)pData->prev_M[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
308  memset(FLATTEN2D(pData->prev_M[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float_complex));
309  pData->prev_Mr[src] = (float**)realloc2d((void**)pData->prev_Mr[src], HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float));
310  memset(FLATTEN2D(pData->prev_Mr[src]), 0, HYBRID_BANDS * (pData->Q)*(pData->Q) * sizeof(float));
311  pData->dirActive[src] = realloc1d(pData->dirActive[src], pData->nGrid * sizeof(int));
312  memset(pData->dirActive[src], 0, pData->nGrid*sizeof(int));
313  }
314  pData->new_M = (float_complex**)realloc2d((void**)pData->new_M, HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float_complex));
315  pData->new_Mr = (float**)realloc2d((void**)pData->new_Mr, HYBRID_BANDS, (pData->Q)*(pData->Q), sizeof(float));
316  pData->interp_M = realloc1d(pData->interp_M, (pData->Q)*(pData->Q) * sizeof(float_complex));
317  pData->interp_Mr = realloc1d(pData->interp_Mr, (pData->Q)*(pData->Q) * sizeof(float));
318  pData->interp_Mr_cmplx = realloc1d(pData->interp_Mr_cmplx, (pData->Q)*(pData->Q) * sizeof(float_complex));
319  memset(pData->interp_Mr_cmplx, 0, (pData->Q)*(pData->Q) * sizeof(float_complex));
320 
321  /* New config */
322  pData->nSources = nSources;
323  pData->procMode = procMode;
324 
325  /* done! */
326  strcpy(pData->progressBarText,"Done!");
327  pData->progressBar0_1 = 1.0f;
328  pData->codecStatus = CODEC_STATUS_INITIALISED;
329 }

331 void spreader_process
332 (
333  void * const hSpr,
334  const float *const * inputs,
335  float ** const outputs,
336  int nInputs,
337  int nOutputs,
338  int nSamples
339 )
340 {
341  spreader_data *pData = (spreader_data*)(hSpr);
342  int q, src, ng, ch, i, j, band, t, nSources, Q, centre_ind, nSpread;
343  float trace, Ey, Eproto, Gcomp;
344  float src_dirs_deg[SPREADER_MAX_NUM_SOURCES][2], src_dir_xyz[3], CprotoDiag[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS], src_spread[MAX_NUM_OUTPUTS];
345  float_complex scaleC, tmp;
346  float_complex tmpFrame[MAX_NUM_CHANNELS][TIME_SLOTS], H_tmp[MAX_NUM_CHANNELS], Cy[MAX_NUM_CHANNELS*MAX_NUM_CHANNELS];
347  float_complex E_dir[MAX_NUM_CHANNELS*MAX_NUM_CHANNELS], V[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS], D[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
348  float_complex Cproto[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
349 #if 0
350  float_complex Cx[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
351  float CxDiag[MAX_NUM_OUTPUTS*MAX_NUM_OUTPUTS];
352 #endif
353  SPREADER_PROC_MODES procMode;
354  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
355 
356  /* copy user parameters to local variables */
357  procMode = pData->procMode;
358  nSources = pData->nSources;
359  Q = pData->Q;
360  memcpy((float*)src_dirs_deg, pData->src_dirs_deg, nSources*2*sizeof(float));
361  memcpy((float*)src_spread, pData->src_spread, nSources*sizeof(float));
362 
363  /* apply binaural panner */
364  if ((nSamples == SPREADER_FRAME_SIZE) && (pData->codecStatus==CODEC_STATUS_INITIALISED) ){
365  pData->procStatus = PROC_STATUS_ONGOING;
366 
367  /* Load time-domain data */
368  for(i=0; i < SAF_MIN(nSources,nInputs); i++)
369  utility_svvcopy(inputs[i], SPREADER_FRAME_SIZE, pData->inputFrameTD[i]);
370  for(; i<nSources; i++)
371  memset(pData->inputFrameTD[i], 0, SPREADER_FRAME_SIZE * sizeof(float));
372 
373  /* Apply time-frequency transform (TFT) */
374  afSTFT_forward_knownDimensions(pData->hSTFT, pData->inputFrameTD, SPREADER_FRAME_SIZE, MAX_NUM_INPUTS, TIME_SLOTS, pData->inputframeTF);
375 
376  /* Zero output buffer */
377  for(band=0; band<HYBRID_BANDS; band++)
378  memset(FLATTEN2D(pData->outputframeTF[band]), 0, Q*TIME_SLOTS*sizeof(float_complex));
379 
380  /* Loop over sources */
381  for(src=0; src<nSources; src++){
382  /* Find the "spread" indices */
383  unitSph2cart(src_dirs_deg[src], 1, 1, src_dir_xyz);
384  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pData->nGrid, 1, 3, 1.0f,
385  pData->grid_dirs_xyz, 3,
386  src_dir_xyz, 1, 0.0f,
387  pData->angles, 1);
388  for(i=0; i<pData->nGrid; i++)
389  pData->angles[i] = acosf(SAF_MIN(pData->angles[i], 0.9999999f))*180.0f/SAF_PI;
390  utility_siminv(pData->angles, pData->nGrid, &centre_ind);
391 
392  /* Define Prototype signals */
393  switch(procMode){
394  case SPREADER_MODE_NAIVE: /* fall through */
395  case SPREADER_MODE_OM:
396  for(band=0; band<HYBRID_BANDS; band++){
397  if(pData->freqVector[band]<MAX_SPREAD_FREQ){
398  /* Loop over all angles, and sum the H_grid's within the spreading area */
399  memset(H_tmp, 0, Q*sizeof(float_complex));
400  for(ng=0,nSpread=0; ng<pData->nGrid; ng++){
401  if(pData->angles[ng] <= (src_spread[src]/2.0f)){
402  for(q=0; q<Q; q++)
403  H_tmp[q] = ccaddf(H_tmp[q], pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + ng]);
404  nSpread++;
405  pData->dirActive[src][ng] = 1;
406  }
407  else
408  pData->dirActive[src][ng] = 0;
409  }
410  }
411  else
412  nSpread = 0;
413 
414  /* If no directions found in the spread area, then just include the nearest one */
415  if(nSpread==0){
416  for(q=0; q<Q; q++)
417  H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
418  nSpread=1;
419  }
420 
421  /* Apply */
422  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, TIME_SLOTS, 1, &calpha,
423  H_tmp, 1,
424  pData->inputframeTF[band][src], TIME_SLOTS, &cbeta,
425  FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS);
426 
427  /* Scale by number of spreading directions */
428  cblas_sscal(/*re+im*/2*Q*TIME_SLOTS, 1.0f/(float)nSpread, (float*)FLATTEN2D(pData->protoframeTF[band]), 1);
429  }
430  break;
431 #if 0
432  case SPREADER_MODE_OM:
433  /* Use the centre direction as the prototype */
434  for(band=0; band<HYBRID_BANDS; band++){
435  for(q=0; q<Q; q++)
436  H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
437  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, TIME_SLOTS, 1, &calpha,
438  H_tmp, 1,
439  pData->inputframeTF[band][src], TIME_SLOTS, &cbeta,
440  FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS);
441  }
442  break;
443 #endif
444 
445  case SPREADER_MODE_EVD:
446  /* Replicate the mono signal for all Q channels */
447  for(band=0; band<HYBRID_BANDS; band++)
448  for(q=0; q<Q; q++)
449  memcpy(pData->protoframeTF[band][q], pData->inputframeTF[band][src], TIME_SLOTS*sizeof(float_complex));
450  break;
451  }
452 
453  /* Main processing */
454  if(procMode==SPREADER_MODE_NAIVE) {
455  /* If naive mode, then we're already done... */
456  for(band=0; band<HYBRID_BANDS; band++)
457  memcpy(FLATTEN2D(pData->spreadframeTF[band]), FLATTEN2D(pData->protoframeTF[band]), Q*TIME_SLOTS*sizeof(float_complex));
458  }
459  else{
460  /* Apply decorrelation of prototype signals */
461  latticeDecorrelator_apply(pData->hDecor[src], pData->protoframeTF, TIME_SLOTS, pData->decorframeTF);
462 
463  /* Compute prototype covariance matrix and average over time */
464  for(band=0; band<HYBRID_BANDS; band++){
465  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, Q, Q, TIME_SLOTS, &calpha,
466  FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS,
467  FLATTEN2D(pData->protoframeTF[band]), TIME_SLOTS, &cbeta,
468  Cproto, Q);
469  cblas_sscal(/*re+im*/2*Q*Q, pData->covAvgCoeff, (float*)pData->Cproto[src][band], 1);
470  cblas_saxpy(/*re+im*/2*Q*Q, 1.0f-pData->covAvgCoeff, (float*)Cproto, 1, (float*)pData->Cproto[src][band], 1);
471  }
472 
473  /* Define target covariance matrices */
474  for(band=0; band<HYBRID_BANDS; band++){
475  /* Sum the H_array outer product matrices for the whole spreading area */
476  if(pData->freqVector[band]<MAX_SPREAD_FREQ){
477  memset(Cy, 0, Q*Q*sizeof(float_complex));
478  memset(H_tmp, 0, Q*sizeof(float_complex));
479  for(ng=0, nSpread=0; ng<pData->nGrid; ng++){
480  if(pData->angles[ng] <= (src_spread[src]/2.0f)){
481  cblas_caxpy(Q*Q, &calpha, pData->HHH[band][ng], 1, Cy, 1);
482  for(q=0; q<Q; q++)
483  H_tmp[q] = ccaddf(H_tmp[q], pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + ng]);
484  nSpread++;
485  pData->dirActive[src][ng] = 1;
486  }
487  else
488  pData->dirActive[src][ng] = 0;
489  }
490  }
491  else
492  nSpread = 0;
493 
494  /* If no directions found in the spread area, then just include the nearest one */
495  if(nSpread==0) {
496  cblas_caxpy(Q*Q, &calpha, pData->HHH[band][centre_ind], 1, Cy, 1);
497  for(q=0; q<Q; q++)
498  H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
499  nSpread++;
500  }
501 #if 1
502  /* Impose target energies too */
503  if (procMode == SPREADER_MODE_OM && pData->freqVector[band]<MAX_SPREAD_FREQ){
504  /* Normalise Cy */
505  trace = 0.0f;
506  for(q=0; q<Q; q++)
507  trace += crealf(Cy[q*Q+q]);
508  cblas_sscal(/*re+im*/2*Q*Q, 1.0f/(trace+2.23e-9f), (float*)Cy, 1);
509 
510  /* Compute signals for the centre of the spread */
511  for(q=0; q<Q; q++)
512  H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
513  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, TIME_SLOTS, 1, &calpha,
514  H_tmp, 1,
515  pData->inputframeTF[band][src], TIME_SLOTS, &cbeta,
516  tmpFrame, TIME_SLOTS);
517 
518  /* Introduce their channel energies into the target */
519  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, Q, Q, TIME_SLOTS, &calpha,
520  tmpFrame, TIME_SLOTS,
521  tmpFrame, TIME_SLOTS, &cbeta,
522  E_dir, Q);
523  trace = 0.0f;
524  for(q=0; q<Q; q++)
525  trace += crealf(E_dir[q*Q+q]);
526  cblas_sscal(/*re+im*/2*Q*Q, trace, (float*)Cy, 1);
527  }
528 #endif
529  /* Average over time */
530  cblas_sscal(/*re+im*/2*Q*Q, pData->covAvgCoeff, (float*)pData->Cy[src][band], 1);
531  cblas_saxpy(/*re+im*/2*Q*Q, 1.0f-pData->covAvgCoeff, (float*)Cy, 1, (float*)pData->Cy[src][band], 1);
532  }
533 
534  /* Formulate mixing matrices */
535  switch(procMode){
536  case SPREADER_MODE_NAIVE: saf_print_error("Shouldn't have gotten this far?"); break;
537  case SPREADER_MODE_EVD:
538  /* For normalising the level of Cy */
539  Ey = Eproto = 0.0f;
540  for(band=0; band<HYBRID_BANDS; band++){
541  for(i=0; i<Q; i++){
542  Ey += crealf(pData->Cy[src][band][i*Q+i]);
543  Eproto += crealf(pData->Cproto[src][band][i*Q+i])+0.000001f;
544  }
545  }
546  Gcomp = sqrtf(Eproto/(Ey+2.23e-9f));
547 
548  /* Compute mixing matrix per band */
549  for(band=0; band<HYBRID_BANDS; band++){
550  memcpy(Cy, pData->Cy[src][band], Q*Q*sizeof(float_complex));
551  cblas_sscal(/*re+im*/2*Q*Q, Gcomp, (float*)Cy, 1);
552  utility_cseig(NULL, Cy, Q, 1, V, D, NULL);
553  for(i=0; i<Q; i++)
554  for(j=0; j<Q; j++)
555  D[i*Q+j] = i==j ? csqrtf(D[i*Q+j]) : cmplxf(0.0f, 0.0f);
556  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, Q, Q, Q, &calpha,
557  V, Q,
558  D, Q, &cbeta,
559  pData->new_M[band], Q);
560  }
561  break;
562 
563  case SPREADER_MODE_OM:
564  for(band=0; band<HYBRID_BANDS; band++){
565  if(pData->freqVector[band]<MAX_SPREAD_FREQ){
566 #if 1
567  /* Diagonalise and diagonally load the Cproto matrices */
568  cblas_ccopy(Q*Q, pData->Cproto[src][band], 1, Cproto, 1);
569  for(i=0; i<Q; i++){
570  for(j=0; j<Q; j++){
571  if(i==j)
572  Cproto[i*Q+i] = craddf(Cproto[i*Q+i], 0.00001f);
573  CprotoDiag[i*Q+j] = i==j ? crealf(Cproto[i*Q+i]) : 0.0f;
574  }
575  }
576 
577  /* Compute mixing matrices */
578  formulate_M_and_Cr_cmplx(pData->hCdf, Cproto, pData->Cy[src][band], pData->Qmix_cmplx, 0, 0.2f, pData->new_M[band], pData->Cr_cmplx);
579  for(i=0; i<Q*Q; i++)
580  pData->Cr[i] = crealf(pData->Cr_cmplx[i]);
581  formulate_M_and_Cr(pData->hCdf_res, CprotoDiag, pData->Cr, pData->Qmix, 0, 0.2f, pData->new_Mr[band], NULL);
582 #else
583  for(q=0; q<Q; q++)
584  H_tmp[q] = pData->H_grid[band*Q*pData->nGrid + q*pData->nGrid + centre_ind];
585  cblas_cgemm(CblasRowMajor, CblasNoTrans, CblasConjTrans, Q, Q, 1, &calpha,
586  H_tmp, 1,
587  H_tmp, 1, &cbeta,
588  Cx, Q);
589  memset(CxDiag, 0, Q*Q*sizeof(float));
590  for(i=0; i<Q; i++)
591  CxDiag[i*Q+i] = crealf(Cx[i*Q+i]);
592 
593  /* Compute mixing matrices */
594  formulate_M_and_Cr_cmplx(pData->hCdf, Cx, pData->Cy[src][band], pData->Qmix_cmplx, 0, 0.2f, pData->new_M[band], pData->Cr_cmplx);
595  for(i=0; i<Q*Q; i++)
596  pData->Cr[i] = crealf(pData->Cr_cmplx[i]);
597  formulate_M_and_Cr(pData->hCdf_res, CxDiag, pData->Cr, pData->Qmix, 0, 0.2f, pData->new_Mr[band], NULL);
598 #endif
599  }
600  else{
601  memcpy(pData->new_M[band], pData->Qmix_cmplx, Q*Q*sizeof(float_complex));
602  memset(pData->new_Mr[band], 0, Q*Q*sizeof(float));
603  }
604  }
605  break;
606  }
607 
608  /* Apply mixing matrices */
609  for(band=0; band<HYBRID_BANDS; band++){
610  for(t=0; t<TIME_SLOTS; t++){
611  scaleC = cmplxf(pData->interpolatorFadeIn[t], 0.0f);
612  utility_cvsmul(pData->new_M[band], &scaleC, Q*Q, pData->interp_M);
613  cblas_saxpy(/*re+im*/2*Q*Q, pData->interpolatorFadeOut[t], (float*)pData->prev_M[src][band], 1, (float*)pData->interp_M, 1);
614  for(i=0; i<Q; i++) {
615  cblas_cdotu_sub(Q, (float_complex*)(&(pData->interp_M[i*Q])), 1,
616  FLATTEN2D((procMode == SPREADER_MODE_EVD ? pData->decorframeTF[band] : pData->protoframeTF[band])) + t,
617  TIME_SLOTS, &(pData->spreadframeTF[band][i][t]));
618  }
619  }
620 
621  /* Also mix in the residual part */
622  if(procMode == SPREADER_MODE_OM){
623  if(pData->freqVector[band]<MAX_SPREAD_FREQ){
624  for(t=0; t<TIME_SLOTS; t++){
625  utility_svsmul(pData->new_Mr[band], &(pData->interpolatorFadeIn[t]), Q*Q, pData->interp_Mr);
626  cblas_saxpy(Q*Q, pData->interpolatorFadeOut[t], pData->prev_Mr[src][band], 1, pData->interp_Mr, 1);
627  cblas_scopy(Q*Q, pData->interp_Mr, 1, (float*)pData->interp_Mr_cmplx, 2);
628  for(i=0; i<Q; i++){
629  cblas_cdotu_sub(Q, (float_complex*)(&(pData->interp_Mr_cmplx[i*Q])), 1, FLATTEN2D(pData->decorframeTF[band]) + t, TIME_SLOTS, &tmp);
630  pData->spreadframeTF[band][i][t] = ccaddf(pData->spreadframeTF[band][i][t], tmp);
631  }
632  }
633  }
634  }
635  }
636  }
637 
638  /* Add the spread frame to the output frame, then move onto the next source... */
639  for(band=0; band<HYBRID_BANDS; band++)
640  cblas_saxpy(/*re+im*/2*Q*TIME_SLOTS, 1.0f, (float*)FLATTEN2D(pData->spreadframeTF[band]), 1, (float*)FLATTEN2D(pData->outputframeTF[band]), 1);
641 
642  /* For next frame */
643  cblas_ccopy(HYBRID_BANDS*Q*Q, FLATTEN2D(pData->new_M), 1, FLATTEN2D(pData->prev_M[src]), 1);
644  cblas_scopy(HYBRID_BANDS*Q*Q, FLATTEN2D(pData->new_Mr), 1, FLATTEN2D(pData->prev_Mr[src]), 1);
645  }
646 
647  /* inverse-TFT */
648  afSTFT_backward_knownDimensions(pData->hSTFT, pData->outputframeTF, SPREADER_FRAME_SIZE, MAX_NUM_OUTPUTS, TIME_SLOTS, pData->outframeTD);
649 
650  /* Copy to output buffer */
651  for (ch = 0; ch < SAF_MIN(Q, nOutputs); ch++)
652  utility_svvcopy(pData->outframeTD[ch], SPREADER_FRAME_SIZE, outputs[ch]);
653  for (; ch < nOutputs; ch++)
654  memset(outputs[ch], 0, SPREADER_FRAME_SIZE*sizeof(float));
655  }
656  else{
657  for (ch=0; ch < nOutputs; ch++)
658  memset(outputs[ch],0, SPREADER_FRAME_SIZE*sizeof(float));
659  }
660 
661  pData->procStatus = PROC_STATUS_NOT_ONGOING;
662 }
