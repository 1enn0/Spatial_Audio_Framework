46 void panner_create
47 (
48  void ** const phPan
49 )
50 {
51  panner_data* pData = (panner_data*)malloc1d(sizeof(panner_data));
52  *phPan = (void*)pData;
53  int ch, dummy;
54 
55  /* default user parameters */
56  panner_loadSourcePreset(SOURCE_CONFIG_PRESET_DEFAULT, pData->src_dirs_deg, &(pData->new_nSources), &(dummy)); /*check setStateInformation if you change default preset*/
57  pData->nSources = pData->new_nSources;
58  pData->DTT = 0.5f;
59  pData->spread_deg = 0.0f;
60  panner_loadLoudspeakerPreset(LOUDSPEAKER_ARRAY_PRESET_STEREO, pData->loudpkrs_dirs_deg, &(pData->new_nLoudpkrs), &(pData->output_nDims)); /*check setStateInformation if you change default preset*/
61  pData->nLoudpkrs = pData->new_nLoudpkrs;
62  pData->yaw = 0.0f;
63  pData->pitch = 0.0f;
64  pData->roll = 0.0f;
65  pData->bFlipYaw = 0;
66  pData->bFlipPitch = 0;
67  pData->bFlipRoll = 0;
68 
69  /* time-frequency transform + buffers */
70  pData->hSTFT = NULL;
71  pData->inputFrameTD = (float**)malloc2d(MAX_NUM_INPUTS, PANNER_FRAME_SIZE, sizeof(float));
72  pData->outputFrameTD = (float**)malloc2d(MAX_NUM_OUTPUTS, PANNER_FRAME_SIZE, sizeof(float));
73  pData->inputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_INPUTS, TIME_SLOTS, sizeof(float_complex));
74  pData->outputframeTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_OUTPUTS, TIME_SLOTS, sizeof(float_complex));
75 
76  /* flags and gain table */
77  pData->progressBar0_1 = 0.0f;
78  pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
79  strcpy(pData->progressBarText,"");
80  pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
81  pData->procStatus = PROC_STATUS_NOT_ONGOING;
82  for(ch=0; ch<MAX_NUM_INPUTS; ch++)
83  pData->recalc_gainsFLAG[ch] = 1;
84  pData->vbap_gtable = NULL;
85  pData->recalc_M_rotFLAG = 1;
86  pData->reInitGainTables = 1;
87 }

89 void panner_destroy
90 (
91  void ** const phPan
92 )
93 {
94  panner_data *pData = (panner_data*)(*phPan);
95 
96  if (pData != NULL) {
97  /* not safe to free memory during intialisation/processing loop */
98  while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
99  pData->procStatus == PROC_STATUS_ONGOING){
100  SAF_SLEEP(10);
101  }
102 
103  /* free afSTFT and buffers */
104  if(pData->hSTFT !=NULL)
105  afSTFT_destroy(&(pData->hSTFT));
106  free(pData->inputFrameTD);
107  free(pData->outputFrameTD);
108  free(pData->inputframeTF);
109  free(pData->outputframeTF);
110  free(pData->vbap_gtable);
111  free(pData->progressBarText);
112 
113  free(pData);
114  pData = NULL;
115  }

118 void panner_init
119 (
120  void * const hPan,
121  int sampleRate
122 )

127  pData->fs = sampleRate;
128  afSTFT_getCentreFreqs(pData->hSTFT, (float)sampleRate, HYBRID_BANDS, pData->freqVector);
129 
130  /* calculate pValue per frequency */
131  getPvalues(pData->DTT, pData->freqVector, HYBRID_BANDS, pData->pValue);
132 
133  /* reinitialise if needed */
134  pData->recalc_M_rotFLAG = 1;

137 void panner_initCodec
138 (
139  void* const hPan
140 )
141 {
142  panner_data *pData = (panner_data*)(hPan);
143 
144  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
145  return; /* re-init not required, or already happening */
146  while (pData->procStatus == PROC_STATUS_ONGOING){
147  /* re-init required, but we need to wait for the current processing loop to end */
148  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
149  SAF_SLEEP(10);
150  }
151 
152  /* for progress bar */
153  pData->codecStatus = CODEC_STATUS_INITIALISING;
154  strcpy(pData->progressBarText,"Initialising");
155  pData->progressBar0_1 = 0.0f;
156 
157  /* reinit TFT if needed */
158  panner_initTFT(hPan);
159 
160  /* reinit gain tables */
161  if(pData->reInitGainTables){
162  panner_initGainTables(hPan);
163  pData->reInitGainTables = 0;
164  }
165 
166  /* done! */
167  strcpy(pData->progressBarText,"Done!");
168  pData->progressBar0_1 = 1.0f;
169  pData->codecStatus = CODEC_STATUS_INITIALISED;
170 
171 }

173 void panner_process
174 (
175  void * const hPan,
176  const float *const * inputs,
177  float ** const outputs,
178  int nInputs,
179  int nOutputs,
180  int nSamples
181 )
182 {
183  panner_data *pData = (panner_data*)(hPan);
184  int t, ch, ls, i, band, nSources, nLoudspeakers, N_azi, aziIndex, elevIndex, idx3d, idx2D;
185  float aziRes, elevRes, pv_f, gains3D_sum_pvf, gains2D_sum_pvf, Rxyz[3][3], hypotxy;
186  float src_dirs[MAX_NUM_INPUTS][2], pValue[HYBRID_BANDS], gains3D[MAX_NUM_OUTPUTS], gains2D[MAX_NUM_OUTPUTS];
187  const float_complex calpha = cmplxf(1.0f, 0.0f), cbeta = cmplxf(0.0f, 0.0f);
188  float_complex outputTemp[MAX_NUM_OUTPUTS][TIME_SLOTS];
189 
190  /* copy user parameters to local variables */
191  memcpy(src_dirs, pData->src_dirs_deg, MAX_NUM_INPUTS*2*sizeof(float));
192  memcpy(pValue, pData->pValue, HYBRID_BANDS*sizeof(float));
193  nSources = pData->nSources;
194  nLoudspeakers = pData->nLoudpkrs;
195 
196  /* apply panner */
197  if ((nSamples == PANNER_FRAME_SIZE) && (pData->vbap_gtable != NULL) && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
198  pData->procStatus = PROC_STATUS_ONGOING;
199 
200  /* Load time-domain data */
201  for(i=0; i < SAF_MIN(nSources,nInputs); i++)
202  utility_svvcopy(inputs[i], PANNER_FRAME_SIZE, pData->inputFrameTD[i]);
203  for(; i<MAX_NUM_INPUTS; i++)
204  memset(pData->inputFrameTD[i], 0, PANNER_FRAME_SIZE * sizeof(float));
205 
206  /* Apply time-frequency transform (TFT) */
207  afSTFT_forward_knownDimensions(pData->hSTFT, pData->inputFrameTD, PANNER_FRAME_SIZE, MAX_NUM_INPUTS, TIME_SLOTS, pData->inputframeTF);
208  memset(FLATTEN3D(pData->outputframeTF), 0, HYBRID_BANDS*MAX_NUM_OUTPUTS*TIME_SLOTS * sizeof(float_complex));
209  memset(outputTemp, 0, MAX_NUM_OUTPUTS*TIME_SLOTS * sizeof(float_complex));
210 
211  /* Rotate source directions */
212  if(pData->recalc_M_rotFLAG){
213  yawPitchRoll2Rzyx (pData->yaw, pData->pitch, pData->roll, 0, Rxyz);
214  for(i=0; i<nSources; i++){
215  pData->src_dirs_xyz[i][0] = cosf(DEG2RAD(pData->src_dirs_deg[i][1])) * cosf(DEG2RAD(pData->src_dirs_deg[i][0]));
216  pData->src_dirs_xyz[i][1] = cosf(DEG2RAD(pData->src_dirs_deg[i][1])) * sinf(DEG2RAD(pData->src_dirs_deg[i][0]));
217  pData->src_dirs_xyz[i][2] = sinf(DEG2RAD(pData->src_dirs_deg[i][1]));
218  pData->recalc_gainsFLAG[i] = 1;
219  }
220  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, nSources, 3, 3, 1.0f,
221  (float*)(pData->src_dirs_xyz), 3,
222  (float*)Rxyz, 3, 0.0f,
223  (float*)(pData->src_dirs_rot_xyz), 3);
224  for(i=0; i<nSources; i++){
225  hypotxy = sqrtf(powf(pData->src_dirs_rot_xyz[i][0], 2.0f) + powf(pData->src_dirs_rot_xyz[i][1], 2.0f));
226  pData->src_dirs_rot_deg[i][0] = RAD2DEG(atan2f(pData->src_dirs_rot_xyz[i][1], pData->src_dirs_rot_xyz[i][0]));
227  pData->src_dirs_rot_deg[i][1] = RAD2DEG(atan2f(pData->src_dirs_rot_xyz[i][2], hypotxy));
228  }
229  pData->recalc_M_rotFLAG = 0;
230  }
231 
232  /* Apply VBAP Panning */
233  if(pData->output_nDims == 3){/* 3-D case */
234  aziRes = (float)pData->vbapTableRes[0];
235  elevRes = (float)pData->vbapTableRes[1];
236  N_azi = (int)(360.0f / aziRes + 0.5f) + 1;
237  for (ch = 0; ch < nSources; ch++) {
238  /* recalculate frequency dependent panning gains */
239  if(pData->recalc_gainsFLAG[ch]){
240  //aziIndex = (int)(matlab_fmodf(pData->src_dirs_deg[ch][0] + 180.0f, 360.0f) / aziRes + 0.5f);
241  //elevIndex = (int)((pData->src_dirs_deg[ch][1] + 90.0f) / elevRes + 0.5f);
242  aziIndex = (int)(matlab_fmodf(pData->src_dirs_rot_deg[ch][0] + 180.0f, 360.0f) / aziRes + 0.5f);
243  elevIndex = (int)((pData->src_dirs_rot_deg[ch][1] + 90.0f) / elevRes + 0.5f);
244  idx3d = elevIndex * N_azi + aziIndex;
245  for (ls = 0; ls < nLoudspeakers; ls++)
246  gains3D[ls] = pData->vbap_gtable[idx3d*nLoudspeakers+ls];
247  for (band = 0; band < HYBRID_BANDS; band++){
248  /* apply pValue per frequency */
249  pv_f = pData->pValue[band];
250  if(pv_f != 2.0f){
251  gains3D_sum_pvf = 0.0f;
252  for (ls = 0; ls < nLoudspeakers; ls++)
253  gains3D_sum_pvf += powf(SAF_MAX(gains3D[ls], 0.0f), pv_f);
254  gains3D_sum_pvf = powf(gains3D_sum_pvf, 1.0f/(pv_f+2.23e-9f));
255  for (ls = 0; ls < nLoudspeakers; ls++)
256  pData->G_src[band][ch][ls] = cmplxf(gains3D[ls] / (gains3D_sum_pvf+2.23e-9f), 0.0f);
257  }
258  else
259  for (ls = 0; ls < nLoudspeakers; ls++)
260  pData->G_src[band][ch][ls] = cmplxf(gains3D[ls], 0.0f);
261  }
262  pData->recalc_gainsFLAG[ch] = 0;
263  }
264  }
265  /* apply panning gains */
266  for (band = 0; band < HYBRID_BANDS; band++) {
267  cblas_cgemm(CblasRowMajor, CblasTrans, CblasNoTrans, nLoudspeakers, TIME_SLOTS, nSources, &calpha,
268  pData->G_src[band], MAX_NUM_OUTPUTS,
269  FLATTEN2D(pData->inputframeTF[band]), TIME_SLOTS, &cbeta,
270  outputTemp, TIME_SLOTS);
271  for (i = 0; i < nLoudspeakers; i++)
272  for (t = 0; t < TIME_SLOTS; t++)
273  pData->outputframeTF[band][i][t] = ccaddf(pData->outputframeTF[band][i][t], outputTemp[i][t]);
274  }
275  }
276  else{/* 2-D case */
277  aziRes = (float)pData->vbapTableRes[0];
278  for (ch = 0; ch < nSources; ch++) {
279  /* recalculate frequency dependent panning gains */
280  if(pData->recalc_gainsFLAG[ch]){
281  //idx2D = (int)((matlab_fmodf(pData->src_dirs_deg[ch][0]+180.0f,360.0f)/aziRes)+0.5f);
282  idx2D = (int)((matlab_fmodf(pData->src_dirs_rot_deg[ch][0]+180.0f,360.0f)/aziRes)+0.5f);
283  for (ls = 0; ls < nLoudspeakers; ls++)
284  gains2D[ls] = pData->vbap_gtable[idx2D*nLoudspeakers+ls];
285  for (band = 0; band < HYBRID_BANDS; band++){
286  /* apply pValue per frequency */
287  pv_f = pData->pValue[band];
288  if(pv_f != 2.0f){
289  gains2D_sum_pvf = 0.0f;
290  for (ls = 0; ls < nLoudspeakers; ls++)
291  gains2D_sum_pvf += powf(SAF_MAX(gains2D[ls], 0.0f), pv_f);
292  gains2D_sum_pvf = powf(gains2D_sum_pvf, 1.0f/(pv_f+2.23e-9f));
293  for (ls = 0; ls < nLoudspeakers; ls++)
294  pData->G_src[band][ch][ls] = cmplxf(gains2D[ls] / (gains2D_sum_pvf+2.23e-9f), 0.0f);
295  }
296  else
297  for (ls = 0; ls < nLoudspeakers; ls++)
298  pData->G_src[band][ch][ls] = cmplxf(gains2D[ls], 0.0f);
299  }
300  pData->recalc_gainsFLAG[ch] = 0;
301  }
302 
303  /* apply panning gains */
304  for (band = 0; band < HYBRID_BANDS; band++){
305  for (ls = 0; ls < nLoudspeakers; ls++)
306  for (t = 0; t < TIME_SLOTS; t++)
307  pData->outputframeTF[band][ls][t] = ccaddf(pData->outputframeTF[band][ls][t], ccmulf(pData->inputframeTF[band][ch][t], pData->G_src[band][ch][ls]));
308  }
309  }
310  }
311 
312  /* scale by sqrt(number of sources) */
313  for (band = 0; band < HYBRID_BANDS; band++)
314  cblas_sscal(/*re+im*/2*nLoudspeakers*TIME_SLOTS, 1.0f/sqrtf((float)nSources), (float*)FLATTEN2D(pData->outputframeTF[band]), 1);
315 
316  /* inverse-TFT and copy to output */
317  afSTFT_backward_knownDimensions(pData->hSTFT, pData->outputframeTF, PANNER_FRAME_SIZE, MAX_NUM_OUTPUTS, TIME_SLOTS, pData->outputFrameTD);
318  for (ch = 0; ch < SAF_MIN(nLoudspeakers, nOutputs); ch++)
319  utility_svvcopy(pData->outputFrameTD[ch], PANNER_FRAME_SIZE, outputs[ch]);
320  for (; ch < nOutputs; ch++)
321  memset(outputs[ch], 0, PANNER_FRAME_SIZE*sizeof(float));
322 
323  }
324  else
325  for (ch=0; ch < nOutputs; ch++)
326  memset(outputs[ch],0, PANNER_FRAME_SIZE*sizeof(float));
327 
328 
329  pData->procStatus = PROC_STATUS_NOT_ONGOING;
330 }

487 void panner_setYaw(void * const hBin, float newYaw)
488 {
