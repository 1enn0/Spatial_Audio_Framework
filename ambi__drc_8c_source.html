43 void ambi_drc_create
44 (
45  void ** const phAmbi
46 )
47 {
48  ambi_drc_data* pData = (ambi_drc_data*)malloc1d(sizeof(ambi_drc_data));
49  *phAmbi = (void*)pData;
50 
51  /* afSTFT stuff and audio buffers*/
52  pData->hSTFT = NULL;
53  pData->frameTD = (float**)malloc2d(MAX_NUM_SH_SIGNALS, AMBI_DRC_FRAME_SIZE, sizeof(float));
54  pData->inputFrameTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
55  pData->outputFrameTF = (float_complex***)malloc3d(HYBRID_BANDS, MAX_NUM_SH_SIGNALS, TIME_SLOTS, sizeof(float_complex));
56 
57  /* internal */
58  pData->fs = 48000;
59 #ifdef ENABLE_TF_DISPLAY
60  pData->gainsTF_bank0 = (float**)malloc2d(HYBRID_BANDS, AMBI_DRC_NUM_DISPLAY_TIME_SLOTS, sizeof(float));
61  pData->gainsTF_bank1 = (float**)malloc2d(HYBRID_BANDS, AMBI_DRC_NUM_DISPLAY_TIME_SLOTS, sizeof(float));
62 #endif
63 
64  /* Default user parameters */
65  pData->theshold = 0.0f;
66  pData->ratio = 8.0f;
67  pData->knee = 0.0f;
68  pData->inGain = 0.0f;
69  pData->outGain = 0.0f;
70  pData->attack_ms = 50.0f;
71  pData->release_ms = 100.0f;
72  pData->chOrdering = CH_ACN;
73  pData->norm = NORM_SN3D;
74  pData->currentOrder = SH_ORDER_FIRST;
75 
76  /* for dynamically allocating the number of channels */
77  ambi_drc_setInputOrder(pData->currentOrder, &(pData->new_nSH));
78  pData->nSH = pData->new_nSH;
79  pData->reInitTFT = 1;
80 }

104 void ambi_drc_init
105 (
106  void * const hAmbi,
107  int sampleRate
108 )
109 {
110  ambi_drc_data *pData = (ambi_drc_data*)(hAmbi);
111  int band;
112 
113  pData->fs = (float)sampleRate;
114  memset(pData->yL_z1, 0, HYBRID_BANDS * sizeof(float));
115  afSTFT_getCentreFreqs(pData->hSTFT, (float)sampleRate, HYBRID_BANDS, pData->freqVector);
116 
117 #ifdef ENABLE_TF_DISPLAY
118  pData->rIdx = 0;
119  pData->wIdx = 1;
120  pData->storeIdx = 0;
121  for (band = 0; band < HYBRID_BANDS; band++) {
122  memset(pData->gainsTF_bank0[band], 0, AMBI_DRC_NUM_DISPLAY_TIME_SLOTS * sizeof(float));
123  memset(pData->gainsTF_bank1[band], 0, AMBI_DRC_NUM_DISPLAY_TIME_SLOTS * sizeof(float));
124  }
125 #endif
126 
127  /* reinitialise if needed */
128  if (pData->reInitTFT == 1) {
129  pData->reInitTFT = 2;
130  ambi_drc_initTFT(hAmbi);
131  pData->reInitTFT = 0;
132  }

135 void ambi_drc_process
136 (
137  void * const hAmbi,
138  const float *const * inputs,
139  float ** const outputs,
140  int nCh,
141  int nSamples
142 )
143 {
144  ambi_drc_data *pData = (ambi_drc_data*)(hAmbi);
145  int i, t, ch, band;
146  float xG, yG, xL, yL, cdB, alpha_a, alpha_r;
147  float makeup, boost, theshold, ratio, knee;
148 
149  /* reinitialise if needed */
150  if(pData->reInitTFT==1){
151  pData->reInitTFT = 2;
152  ambi_drc_initTFT(hAmbi);
153  pData->reInitTFT = 0;
154  }
155 
156  /* local copies of user parameters */
157  alpha_a = expf(-1.0f / ( (pData->attack_ms / ((float)AMBI_DRC_FRAME_SIZE / (float)TIME_SLOTS)) * pData->fs * 0.001f));
158  alpha_r = expf(-1.0f / ( (pData->release_ms / ((float)AMBI_DRC_FRAME_SIZE / (float)TIME_SLOTS)) * pData->fs * 0.001f));
159  boost = powf(10.0f, pData->inGain / 20.0f);
160  makeup = powf(10.0f, pData->outGain / 20.0f);
161  theshold = pData->theshold;
162  ratio = pData->ratio;
163  knee = pData->knee;
164 
165  /* Main processing loop */
166  if (nSamples == AMBI_DRC_FRAME_SIZE && pData->reInitTFT == 0) {
167 
168  /* Load time-domain data */
169  for(i=0; i < SAF_MIN(pData->nSH, nCh); i++)
170  utility_svvcopy(inputs[i], AMBI_DRC_FRAME_SIZE, pData->frameTD[i]);
171  for(; i<pData->nSH; i++)
172  memset(pData->frameTD[i], 0, AMBI_DRC_FRAME_SIZE * sizeof(float));
173 
174  /* Apply time-frequency transform */
175  afSTFT_forward_knownDimensions(pData->hSTFT, pData->frameTD, AMBI_DRC_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->inputFrameTF);
176 
177  /* Main processing: */
178  /* Calculate the dynamic range compression gain factors per frequency band based on the omnidirectional component.
179  * McCormack, L., & Välimäki, V. (2017). "FFT-Based Dynamic Range Compression". in Proceedings of the 14th
180  * Sound and Music Computing Conference, July 5-8, Espoo, Finland.*/
181  for (t = 0; t < TIME_SLOTS; t++) {
182  for (band = 0; band < HYBRID_BANDS; band++) {
183  /* apply input boost */
184  for (ch = 0; ch < pData->nSH; ch++)
185  pData->inputFrameTF[band][ch][t] = crmulf(pData->inputFrameTF[band][ch][t], boost);
186 
187  /* calculate gain factor for this frequency based on the omni component */
188  xG = 10.0f*log10f(powf(cabsf(pData->inputFrameTF[band][0/* omni */][t]), 2.0f) + 2e-13f);
189  yG = ambi_drc_gainComputer(xG, theshold, ratio, knee);
190  xL = xG - yG;
191  yL = ambi_drc_smoothPeakDetector(xL, pData->yL_z1[band], alpha_a, alpha_r);
192  pData->yL_z1[band] = yL;
193  cdB = -yL;
194  cdB = SAF_MAX(AMBI_DRC_SPECTRAL_FLOOR, sqrtf(powf(10.0f, cdB / 20.0f)));
195 
196 #ifdef ENABLE_TF_DISPLAY
197  /* store gain factors in circular buffer for plotting */
198  if(pData->storeIdx==0)
199  pData->gainsTF_bank0[band][pData->wIdx] = cdB;
200  else
201  pData->gainsTF_bank1[band][pData->wIdx] = cdB;
202 #endif
203  /* apply same gain factor to all SH components, the spatial characteristics will be preserved
204  * (although, ones perception of them may of course change) */
205  for (ch = 0; ch < pData->nSH; ch++)
206  pData->outputFrameTF[band][ch][t] = crmulf(pData->inputFrameTF[band][ch][t], cdB*makeup);
207  }
208 #ifdef ENABLE_TF_DISPLAY
209  /* increment circular buffer indices */
210  pData->wIdx++;
211  pData->rIdx++;
212  if (pData->wIdx >= AMBI_DRC_NUM_DISPLAY_TIME_SLOTS){
213  pData->wIdx = 0;
214  pData->storeIdx = pData->storeIdx == 0 ? 1 : 0;
215  }
216  if (pData->rIdx >= AMBI_DRC_NUM_DISPLAY_TIME_SLOTS)
217  pData->rIdx = 0;
218 #endif
219  }
220 
221  /* Inverse time-frequency transform */
222  afSTFT_backward_knownDimensions(pData->hSTFT, pData->outputFrameTF, AMBI_DRC_FRAME_SIZE, MAX_NUM_SH_SIGNALS, TIME_SLOTS, pData->frameTD);
223 
224  /* Copy to output */
225  for(ch = 0; ch < SAF_MIN(pData->nSH, nCh); ch++)
226  utility_svvcopy(pData->frameTD[ch], AMBI_DRC_FRAME_SIZE, outputs[ch]);
227  for (; ch < nCh; ch++)
228  memset(outputs[ch], 0, AMBI_DRC_FRAME_SIZE*sizeof(float));
229  }
230  else {
231  for (ch=0; ch < nCh; ch++)
232  memset(outputs[ch], 0, AMBI_DRC_FRAME_SIZE*sizeof(float));
233  }
234 }
