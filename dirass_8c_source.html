39 void dirass_create
40 (
41  void ** const phDir
42 )
43 {
44  dirass_data* pData = (dirass_data*)malloc1d(sizeof(dirass_data));
45  *phDir = (void*)pData;
46  int i;
47 
48  /* Default user parameters */
49  pData->inputOrder = pData->new_inputOrder = SH_ORDER_FIRST;
50  pData->beamType = STATIC_BEAM_TYPE_HYPERCARDIOID;
51  pData->DirAssMode = REASS_UPSCALE;
52  pData->upscaleOrder = pData->new_upscaleOrder = UPSCALE_ORDER_TENTH;
53  pData->gridOption = GRID_GEOSPHERE_8;
54  pData->pmapAvgCoeff = 0.666f;
55  pData->minFreq_hz = 100.0f;
56  pData->maxFreq_hz = 8e3f;
57  pData->dispWidth = 120;
58  pData->chOrdering = CH_ACN;
59  pData->norm = NORM_SN3D;
60  pData->HFOVoption = HFOV_360;
61  pData->aspectRatioOption = ASPECT_RATIO_2_1;
62 
63  /* codec data */
64  pData->pars = (dirass_codecPars*)malloc1d(sizeof(dirass_codecPars));
65  dirass_codecPars* pars = pData->pars;
66  pars->interp_dirs_deg = NULL;
67  pars->interp_dirs_rad = NULL;
68  pars->Y_up = NULL;
69  pars->interp_table = NULL;
70  pars->w = NULL;
71  pars->Cw = NULL;
72  pars->Uw = NULL;
73  pars->Cxyz = NULL;
74  pars->ss = NULL;
75  pars->ssxyz = NULL;
76  pars->est_dirs = NULL;
77  pars->est_dirs_idx = NULL;
78  pars->prev_intensity = NULL;
79  pars->prev_energy = NULL;
80 
81  /* internal */
82  pData->progressBar0_1 = 0.0f;
83  pData->progressBarText = malloc1d(PROGRESSBARTEXT_CHAR_LENGTH*sizeof(char));
84  strcpy(pData->progressBarText,"");
85  pData->codecStatus = CODEC_STATUS_NOT_INITIALISED;
86  pData->procStatus = PROC_STATUS_NOT_ONGOING;
87 
88  /* display */
89  pData->pmap = NULL;
90  for(i=0; i<NUM_DISP_SLOTS; i++)
91  pData->pmap_grid[i] = NULL;
92  pData->pmapReady = 0;
93  pData->recalcPmap = 1;
94 
95  /* set FIFO buffers */
96  pData->FIFO_idx = 0;
97  memset(pData->inFIFO, 0, MAX_NUM_INPUT_SH_SIGNALS*DIRASS_FRAME_SIZE*sizeof(float));
98 }

100 void dirass_destroy
101 (
102  void ** const phDir
103 )
104 {
105  dirass_data *pData = (dirass_data*)(*phDir);
106  dirass_codecPars* pars;
107  int i;
108 
109  if (pData != NULL) {
110  /* not safe to free memory during intialisation/processing loop */
111  while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
112  pData->procStatus == PROC_STATUS_ONGOING){
113  SAF_SLEEP(10);

133  free(pars->est_dirs_idx);
134  free(pars->prev_intensity);
135  free(pars->prev_energy);
136 
137  free(pData->pars);
138  free(pData->progressBarText);

166 void dirass_initCodec
167 (
168  void* const hDir
169 )
170 {
171  dirass_data *pData = (dirass_data*)(hDir);
172 
173  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
174  return; /* re-init not required, or already happening */
175  while (pData->procStatus == PROC_STATUS_ONGOING){
176  /* re-init required, but we need to wait for the current processing loop to end */
177  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
178  SAF_SLEEP(10);
179  }
180 
181  /* for progress bar */
182  pData->codecStatus = CODEC_STATUS_INITIALISING;
183  strcpy(pData->progressBarText,"Initialising");
184  pData->progressBar0_1 = 0.0f;
185 
186  dirass_initAna(hDir);
187 
188  /* done! */
189  strcpy(pData->progressBarText,"Done!");
190  pData->progressBar0_1 = 1.0f;
191  pData->codecStatus = CODEC_STATUS_INITIALISED;
192 }

194 void dirass_analysis
195 (
196  void * const hDir,
197  const float *const * inputs,
198  int nInputs,
199  int nSamples,
200  int isPlaying
201 )
202 {
203  dirass_data *pData = (dirass_data*)(hDir);
204  dirass_codecPars* pars = pData->pars;
205  int s, i, j, k, ch, sec_nSH, secOrder, nSH, up_nSH;
206  float intensity[3];
207 
208  /* local copy of user parameters */
209  int inputOrder, DirAssMode, upscaleOrder;
210  float pmapAvgCoeff, minFreq_hz, maxFreq_hz;
211  NORM_TYPES norm;
212  CH_ORDER chOrdering;
213  norm = pData->norm;
214  chOrdering = pData->chOrdering;
215  pmapAvgCoeff = pData->pmapAvgCoeff;
216  DirAssMode = pData->DirAssMode;
217  upscaleOrder = pData->upscaleOrder;
218  minFreq_hz = pData->minFreq_hz;
219  maxFreq_hz = pData->maxFreq_hz;
220  inputOrder = pData->inputOrder;
221  secOrder = inputOrder-1;
222  nSH = (inputOrder+1)*(inputOrder+1);
223  sec_nSH = (secOrder+1)*(secOrder+1);
224  up_nSH = (upscaleOrder+1)*(upscaleOrder+1);
225 
226  /* Loop over all samples */
227  for(s=0; s<nSamples; s++){
228  /* Load input signals into inFIFO buffer */
229  for(ch=0; ch<SAF_MIN(nInputs,nSH); ch++)
230  pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
231  for(; ch<nSH; ch++) /* Zero any channels that were not given */
232  pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
233 
234  /* Increment buffer index */
235  pData->FIFO_idx++;
236 
237  /* Process frame if inFIFO is full and codec is ready for it */
238  if (pData->FIFO_idx >= DIRASS_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) && isPlaying) {
239  pData->FIFO_idx = 0;
240  pData->procStatus = PROC_STATUS_ONGOING;
241 
242  /* Load time-domain data */
243  for(ch=0; ch<nSH; ch++)
244  memcpy(pData->SHframeTD[ch],pData->inFIFO[ch], DIRASS_FRAME_SIZE*sizeof(float));
245 
246  /* account for input channel order */
247  switch(chOrdering){
248  case CH_ACN: /* already ACN */ break; /* Otherwise, convert to ACN... */
249  case CH_FUMA: convertHOAChannelConvention((float*)pData->SHframeTD, inputOrder, DIRASS_FRAME_SIZE, HOA_CH_ORDER_FUMA, HOA_CH_ORDER_ACN); break;
250  }
251 
252  /* account for input normalisation scheme */
253  switch(norm){
254  case NORM_N3D: /* already in N3D, do nothing */ break; /* Otherwise, convert to N3D... */
255  case NORM_SN3D: convertHOANormConvention((float*)pData->SHframeTD, inputOrder, DIRASS_FRAME_SIZE, HOA_NORM_SN3D, HOA_NORM_N3D); break;
256  case NORM_FUMA: convertHOANormConvention((float*)pData->SHframeTD, inputOrder, DIRASS_FRAME_SIZE, HOA_NORM_FUMA, HOA_NORM_N3D); break;
257  }
258 
259  /* update the dirass powermap */
260  if(pData->recalcPmap==1){
261  pData->recalcPmap = 0;
262  pData->pmapReady = 0;
263 
264  /* filter input signals */
265  float b[3], a[3];
266  biQuadCoeffs(BIQUAD_FILTER_HPF, minFreq_hz, pData->fs, 0.7071f, 0.0f, b, a);
267  for(i=0; i<nSH; i++)
268  applyBiQuadFilter(b, a, pData->Wz12_hpf[i], pData->SHframeTD[i], DIRASS_FRAME_SIZE);
269  biQuadCoeffs(BIQUAD_FILTER_LPF, maxFreq_hz, pData->fs, 0.7071f, 0.0f, b, a);
270  for(i=0; i<nSH; i++)
271  applyBiQuadFilter(b, a, pData->Wz12_lpf[i], pData->SHframeTD[i], DIRASS_FRAME_SIZE);
272 
273  /* DoA estimation for each spatially-localised sector */
274  if(DirAssMode==REASS_UPSCALE || DirAssMode==REASS_NEAREST){
275  /* Beamform using the sector patterns */
276  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->grid_nDirs, DIRASS_FRAME_SIZE, sec_nSH, 1.0f,
277  pars->Cw, sec_nSH,
278  (const float*)pData->SHframeTD, DIRASS_FRAME_SIZE, 0.0f,
279  pars->ss, DIRASS_FRAME_SIZE);
280 
281  for(i=0; i<pars->grid_nDirs; i++){
282  /* beamforming to get velocity patterns */
283  cblas_sgemm(CblasRowMajor, CblasTrans, CblasNoTrans, 3, DIRASS_FRAME_SIZE, nSH, 1.0f,
284  &(pars->Cxyz[i*nSH*3]), 3,
285  (const float*)pData->SHframeTD, DIRASS_FRAME_SIZE, 0.0f,
286  pars->ssxyz, DIRASS_FRAME_SIZE);
287 
288  /* take the sum or mean ss.*ssxyz, to get intensity vector */
289  memset(intensity, 0, 3*sizeof(float));
290  for(k=0; k<3; k++){
291  for(j=0; j<DIRASS_FRAME_SIZE; j++)
292  intensity[k] += pars->ssxyz[k*DIRASS_FRAME_SIZE + j] * pars->ss[i*DIRASS_FRAME_SIZE+j];
293  intensity[k] /= (float)DIRASS_FRAME_SIZE;
294 
295  /* average over time */
296  intensity[k] = pmapAvgCoeff * (pars->prev_intensity[i*3+k]) + (1.0f-pmapAvgCoeff) * intensity[k];
297  pars->prev_intensity[i*3+k] = intensity[k];
298  }
299 
300  /* extract DoA [azi elev] convention */
301  pars->est_dirs[i*2] = atan2f(intensity[1], intensity[0]);
302  pars->est_dirs[i*2+1] = atan2f(intensity[2], sqrtf(powf(intensity[0], 2.0f) + powf(intensity[1], 2.0f)));
303  if(DirAssMode==REASS_UPSCALE)
304  pars->est_dirs[i*2+1] = M_PI/2.0f - pars->est_dirs[i*2+1]; /* convert to inclination */
305  }
306  }
307 
308  /* Obtain pmap/upscaled pmap in the case of REASS_MODE_OFF and REASS_UPSCALE modes, respectively.
309  * OR find the nearest display grid indices, corresponding to the DoA estimates, for the REASS_NEAREST mode */
310  switch(DirAssMode) {
311  default:
312  case REASS_MODE_OFF:
313  /* Standard beamformer-based pmap */
314  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->grid_nDirs, DIRASS_FRAME_SIZE, nSH, 1.0f,
315  pars->w, nSH,
316  (const float*)pData->SHframeTD, DIRASS_FRAME_SIZE, 0.0f,
317  pars->ss, DIRASS_FRAME_SIZE);
318 
319  /* sum energy over the length of the frame to obtain the pmap */
320  memset(pData->pmap, 0, pars->grid_nDirs *sizeof(float));
321  for(i=0; i<pars->grid_nDirs; i++)
322  for(j=0; j<DIRASS_FRAME_SIZE; j++)
323  pData->pmap[i] += (pars->ss[i*DIRASS_FRAME_SIZE+j])*(pars->ss[i*DIRASS_FRAME_SIZE+j]);
324 
325  /* average energy over time */
326  for(i=0; i<pars->grid_nDirs; i++){
327  pData->pmap[i] = pmapAvgCoeff * (pars->prev_energy[i]) + (1.0f-pmapAvgCoeff) * (pData->pmap[i]);
328  pars->prev_energy[i] = pData->pmap[i];
329  }
330 
331  /* interpolate the pmap */
332  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->interp_nDirs, 1, pars->grid_nDirs, 1.0f,
333  pars->interp_table, pars->grid_nDirs,
334  pData->pmap, 1, 0.0f,
335  pData->pmap_grid[pData->dispSlotIdx], 1);
336  break;
337 
338  case REASS_UPSCALE:
339  /* upscale */
340  getSHreal_recur(upscaleOrder, pars->est_dirs, pars->grid_nDirs, pars->Y_up);
341  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, up_nSH, DIRASS_FRAME_SIZE, pars->grid_nDirs, 1.0f,
342  pars->Y_up, pars->grid_nDirs,
343  pars->ss, DIRASS_FRAME_SIZE, 0.0f,
344  (float*)pData->SHframe_upTD, DIRASS_FRAME_SIZE);
345 
346  /* Beamform using the new spatially upscaled frame */
347  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->grid_nDirs, DIRASS_FRAME_SIZE, up_nSH, 1.0f,
348  pars->Uw, up_nSH,
349  (float*)pData->SHframe_upTD, DIRASS_FRAME_SIZE, 0.0f,
350  pars->ss, DIRASS_FRAME_SIZE);
351 
352  /* sum energy over the length of the frame to obtain the pmap */
353  memset(pData->pmap, 0, pars->grid_nDirs *sizeof(float));
354  for(i=0; i<pars->grid_nDirs; i++)
355  for(j=0; j<DIRASS_FRAME_SIZE; j++)
356  pData->pmap[i] += (pars->ss[i*DIRASS_FRAME_SIZE+j])*(pars->ss[i*DIRASS_FRAME_SIZE+j]);
357 
358  /* average energy over time */
359  for(i=0; i<pars->grid_nDirs; i++){
360  pData->pmap[i] = pmapAvgCoeff * (pars->prev_energy[i]) + (1.0f-pmapAvgCoeff) * (pData->pmap[i]);
361  pars->prev_energy[i] = pData->pmap[i];
362  }
363 
364  /* interpolate the pmap */
365  cblas_sgemm(CblasRowMajor, CblasNoTrans, CblasNoTrans, pars->interp_nDirs, 1, pars->grid_nDirs, 1.0f,
366  pars->interp_table, pars->grid_nDirs,
367  pData->pmap, 1, 0.0f,
368  pData->pmap_grid[pData->dispSlotIdx], 1);
369  break;
370 
371  case REASS_NEAREST:
372  /* Assign the sector energies to the nearest display grid point */
373  findClosestGridPoints(pars->interp_dirs_rad, pars->interp_nDirs, pars->est_dirs, pars->grid_nDirs, 0, pars->est_dirs_idx, NULL, NULL);
374  memset(pData->pmap_grid[pData->dispSlotIdx], 0, pars->interp_nDirs * sizeof(float));
375  for(i=0; i< pars->grid_nDirs; i++)
376  for(j=0; j<DIRASS_FRAME_SIZE; j++)
377  pData->pmap[i] = (pars->ss[i*DIRASS_FRAME_SIZE+j])*(pars->ss[i*DIRASS_FRAME_SIZE+j]);
378 
379  /* average energy over time, and assign to nearest grid direction */
380  for(i=0; i<pars->grid_nDirs; i++){
381  pData->pmap[i] = pmapAvgCoeff * (pars->prev_energy[i]) + (1.0f-pmapAvgCoeff) * (pData->pmap[i]);
382  pars->prev_energy[i] = pData->pmap[i];
383  pData->pmap_grid[pData->dispSlotIdx][pars->est_dirs_idx[i]] += pData->pmap[i];
384  }
385  break;
386  }
387 
388  /* ascertain the minimum and maximum values for pmap colour scaling */
389  int ind;
390  utility_siminv(pData->pmap_grid[pData->dispSlotIdx], pars->interp_nDirs, &ind);
391  pData->pmap_grid_minVal = pData->pmap_grid[pData->dispSlotIdx][ind];
392  utility_simaxv(pData->pmap_grid[pData->dispSlotIdx], pars->interp_nDirs, &ind);
393  pData->pmap_grid_maxVal = pData->pmap_grid[pData->dispSlotIdx][ind];
394 
395  /* normalise the pmap to 0..1 */
396  for(i=0; i<pars->interp_nDirs; i++)
397  pData->pmap_grid[pData->dispSlotIdx][i] = (pData->pmap_grid[pData->dispSlotIdx][i]-pData->pmap_grid_minVal)/(pData->pmap_grid_maxVal-pData->pmap_grid_minVal+1e-11f);
398 
399  /* signify that the pmap in the current slot is ready for plotting */
400  pData->dispSlotIdx++;
401  if(pData->dispSlotIdx>=NUM_DISP_SLOTS)
402  pData->dispSlotIdx = 0;
403  pData->pmapReady = 1;
404  }
405  }
406  else if(pData->FIFO_idx >= DIRASS_FRAME_SIZE){
407  /* reset FIFO_idx index if codec was not ready */
408  pData->FIFO_idx = 0;
409  }
410  }
411 
412  pData->procStatus = PROC_STATUS_NOT_ONGOING;
413 }
