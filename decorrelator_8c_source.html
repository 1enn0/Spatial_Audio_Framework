65 )
66 {
67  decorrelator_data *pData = (decorrelator_data*)(*phDecor);
68 
69  if (pData != NULL) {
70  /* not safe to free memory during intialisation/processing loop */
71  while (pData->codecStatus == CODEC_STATUS_INITIALISING ||
72  pData->procStatus == PROC_STATUS_ONGOING){
73  SAF_SLEEP(10);

75 
76  /* free afSTFT and buffers */

78  afSTFT_destroy(&(pData->hSTFT));

85 
86  transientDucker_destroy(&(pData->hDucker));
87  latticeDecorrelator_destroy(&(pData->hDecor));
88 
89  free(pData);
90  pData = NULL;

110 void decorrelator_initCodec
111 (
112  void* const hDecor
113 )
114 {
115  decorrelator_data *pData = (decorrelator_data*)(hDecor);
116  int nChannels;
117 
118  if (pData->codecStatus != CODEC_STATUS_NOT_INITIALISED)
119  return; /* re-init not required, or already happening */
120  while (pData->procStatus == PROC_STATUS_ONGOING){
121  /* re-init required, but we need to wait for the current processing loop to end */
122  pData->codecStatus = CODEC_STATUS_INITIALISING; /* indicate that we want to init */
123  SAF_SLEEP(10);
124  }
125 
126  /* for progress bar */
127  pData->codecStatus = CODEC_STATUS_INITIALISING;
128  strcpy(pData->progressBarText,"Preparing Decorrelators");
129  pData->progressBar0_1 = 0.0f;
130 
131  /* (Re)Initialise afSTFT */
132  nChannels = pData->new_nChannels;
133  if(pData->hSTFT==NULL)
134  afSTFT_create(&(pData->hSTFT), nChannels, nChannels, HOP_SIZE, 0, 1, AFSTFT_BANDS_CH_TIME);
135  else if(pData->nChannels != nChannels) {/* Or change the number of channels */
136  afSTFT_channelChange(pData->hSTFT, nChannels, nChannels);
137  afSTFT_clearBuffers(pData->hSTFT);
138  }
139  pData->nChannels = nChannels;
140 
141  /* Init transient ducker */
142  transientDucker_destroy(&(pData->hDucker));
143  transientDucker_create(&(pData->hDucker), nChannels, HYBRID_BANDS);
144 
145  /* Init decorrelator */
146  const int orders[4] = {20, 15, 6, 3}; /* 20th order up to 700Hz, 15th->2.4kHz, 6th->4kHz, 3rd->12kHz, NONE(only delays)->Nyquist */
147  const float freqCutoffs[4] = {600.0f, 2.4e3f, 4.0e3f, 12e3f};
148  //const float freqCutoffs[4] = {900.0f, 6.8e3f, 12e3f, 16e3f};
149  const int maxDelay = 8;
150  latticeDecorrelator_destroy(&(pData->hDecor));
151  latticeDecorrelator_create(&(pData->hDecor), pData->fs, HOP_SIZE, pData->freqVector, HYBRID_BANDS, pData->nChannels, (int*)orders, (float*)freqCutoffs, 4, maxDelay, 0, 0.75f);
152 
153  /* done! */
154  strcpy(pData->progressBarText,"Done!");
155  pData->progressBar0_1 = 1.0f;
156  pData->codecStatus = CODEC_STATUS_INITIALISED;
157 }

159 void decorrelator_process
160 (
161  void * const hDecor,
162  const float *const * inputs,
163  float ** const outputs,
164  int nInputs,
165  int nOutputs,
166  int nSamples
167 )
168 {
169  decorrelator_data *pData = (decorrelator_data*)(hDecor);
170  int ch, i, band, enableTransientDucker, compensateLevel;
171  float decorAmount;
172 
173  /* local copies of user parameters */
174  int nCH;
175  nCH = pData->nChannels;
176  decorAmount = pData->decorAmount;
177  enableTransientDucker = pData->enableTransientDucker;
178  compensateLevel = pData->compensateLevel;
179 
180  /* Process frame */
181  if (nSamples == DECORRELATOR_FRAME_SIZE && (pData->codecStatus == CODEC_STATUS_INITIALISED) ) {
182  pData->procStatus = PROC_STATUS_ONGOING;
183 
184  /* Load time-domain data */
185  for(i=0; i < SAF_MIN(nCH, nInputs); i++)
186  utility_svvcopy(inputs[i], DECORRELATOR_FRAME_SIZE, pData->InputFrameTD[i]);
187  for(; i<nCH; i++)
188  memset(pData->InputFrameTD[i], 0, DECORRELATOR_FRAME_SIZE * sizeof(float)); /* fill remaining channels with zeros */
189 
190  /* Apply time-frequency transform (TFT) */
191  afSTFT_forward_knownDimensions(pData->hSTFT, pData->InputFrameTD, DECORRELATOR_FRAME_SIZE, MAX_NUM_CHANNELS, TIME_SLOTS, pData->InputFrameTF);
192 
193  /* Apply decorrelation */
194  if(enableTransientDucker){
195  /* remove transients */
196  transientDucker_apply(pData->hDucker, pData->InputFrameTF, TIME_SLOTS, 0.95f, 0.995f, pData->OutputFrameTF, pData->transientFrameTF);
197  /* decorrelate only the residual */
198  latticeDecorrelator_apply(pData->hDecor, pData->OutputFrameTF, TIME_SLOTS, pData->OutputFrameTF);
199  }
200  else
201  latticeDecorrelator_apply(pData->hDecor, pData->InputFrameTF, TIME_SLOTS, pData->OutputFrameTF);
202 
203  /* Optionally compensate for the level (as they channels wll no longer sum coherently) */
204  if(compensateLevel){
205  for(band=0; band<HYBRID_BANDS; band++)
206  cblas_sscal(/*re+im*/2*nCH*TIME_SLOTS, 0.75f*(float)nCH/(sqrtf((float)nCH)), (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
207  }
208 
209  /* re-introduce the transient part */
210  if(enableTransientDucker){
211  //scalec = cmplxf(1.0f, 0.0f);//!compensateLevel ? cmplxf(1.25f*(sqrtf((float)nCH)/(float)nCH), 0.0f) : cmplxf(1.0f, 0.0f);
212  for(band=0; band<HYBRID_BANDS; band++)
213  cblas_saxpy(/*re+im*/2*nCH*TIME_SLOTS, 1.0f, (float*)FLATTEN2D(pData->transientFrameTF[band]), 1, (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
214  }
215 
216  /* Mix thedecorrelated audio with the input non-decorrelated audio */
217  for(band=0; band<HYBRID_BANDS; band++){
218  cblas_sscal(/*re+im*/2*nCH*TIME_SLOTS, decorAmount, (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
219  cblas_saxpy(/*re+im*/2*nCH*TIME_SLOTS, 1.0f-decorAmount, (float*)FLATTEN2D(pData->InputFrameTF[band]), 1, (float*)FLATTEN2D(pData->OutputFrameTF[band]), 1);
220  }
221 
222  /* inverse-TFT */
223  afSTFT_backward_knownDimensions(pData->hSTFT, pData->OutputFrameTF, DECORRELATOR_FRAME_SIZE, MAX_NUM_CHANNELS, TIME_SLOTS, pData->OutputFrameTD);
224 
225  /* Copy to output buffer */
226  for (ch = 0; ch < SAF_MIN(nCH, nOutputs); ch++)
227  utility_svvcopy(pData->OutputFrameTD[ch], DECORRELATOR_FRAME_SIZE, outputs[ch]);
228  for (; ch < nOutputs; ch++)
229  memset(outputs[ch], 0, DECORRELATOR_FRAME_SIZE*sizeof(float));
230  }
231  else
232  for (ch=0; ch < nOutputs; ch++)
233  memset(outputs[ch],0, DECORRELATOR_FRAME_SIZE*sizeof(float));
234 
235  pData->procStatus = PROC_STATUS_NOT_ONGOING;
236 }
