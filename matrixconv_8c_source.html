95 void matrixconv_process
96 (
97  void * const hMCnv,
98  const float *const * inputs,
99  float ** const outputs,
100  int nInputs,
101  int nOutputs,
102  int nSamples
103 )
104 {
105  matrixconv_data *pData = (matrixconv_data*)(hMCnv);
106  int s, ch, i;
107  int numInputChannels, numOutputChannels;
108 
109  matrixconv_checkReInit(hMCnv);
110 
111  /* prep */
112  numInputChannels = pData->nInputChannels;
113  numOutputChannels = pData->nOutputChannels;
114 
115  for(s=0; s<nSamples; s++){
116  /* Load input signals into inFIFO buffer */
117  for(ch=0; ch<SAF_MIN(SAF_MIN(nInputs,numInputChannels),MAX_NUM_CHANNELS); ch++)
118  pData->inFIFO[ch][pData->FIFO_idx] = inputs[ch][s];
119  for(; ch<numInputChannels; ch++) /* Zero any channels that were not given */
120  pData->inFIFO[ch][pData->FIFO_idx] = 0.0f;
121 
122  /* Pull output signals from outFIFO buffer */
123  for(ch=0; ch<SAF_MIN(SAF_MIN(nOutputs, numOutputChannels),MAX_NUM_CHANNELS); ch++)
124  outputs[ch][s] = pData->outFIFO[ch][pData->FIFO_idx];
125  for(; ch<nOutputs; ch++) /* Zero any extra channels */
126  outputs[ch][s] = 0.0f;
127 
128  /* Increment buffer index */
129  pData->FIFO_idx++;
130 
131  /* Process frame if inFIFO is full and filters are loaded and saf_matrixConv_apply is ready for it */
132  if (pData->FIFO_idx >= pData->hostBlockSize_clamped && pData->reInitFilters == 0 ) {
133  pData->FIFO_idx = 0;
134 
135  /* Load time-domain data */
136  for(i=0; i < numInputChannels; i++)
137  utility_svvcopy(pData->inFIFO[i], pData->hostBlockSize_clamped, pData->inputFrameTD[i]);
138 
139  /* Apply matrix convolution */
140  if(pData->hMatrixConv != NULL && pData->filter_length>0)
141  saf_matrixConv_apply(pData->hMatrixConv, FLATTEN2D(pData->inputFrameTD), FLATTEN2D(pData->outputFrameTD));
142  /* if the matrix convolver handle has not been initialised yet (i.e. no filters have been loaded) then zero the output */
143  else
144  memset(FLATTEN2D(pData->outputFrameTD), 0, MAX_NUM_CHANNELS * (pData->hostBlockSize_clamped)*sizeof(float));
145 
146  /* copy signals to output buffer */
147  for (i = 0; i < SAF_MIN(numOutputChannels, MAX_NUM_CHANNELS); i++)
148  utility_svvcopy(pData->outputFrameTD[i], pData->hostBlockSize_clamped, pData->outFIFO[i]);
149  }
150  else if(pData->FIFO_idx >= pData->hostBlockSize_clamped){
151  /* clear outFIFO if codec was not ready */
152  pData->FIFO_idx = 0;
153  memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
154  }
155  }
156 }

167 void matrixconv_checkReInit(void* const hMCnv)
168 {
169  matrixconv_data *pData = (matrixconv_data*)(hMCnv);
170 
171  /* reinitialise if needed */
172  if ((pData->reInitFilters == 1) && (pData->filters != NULL)) {
173  pData->reInitFilters = 2;
174  saf_matrixConv_destroy(&(pData->hMatrixConv));
175  pData->hMatrixConv = NULL;
176 
177  /* if length of the loaded wav file was not divisable by the specified number of inputs, then the handle remains NULL,
178  * and no convolution is applied */
179  pData->hostBlockSize_clamped = SAF_CLAMP(pData->hostBlockSize, MIN_FRAME_SIZE, MAX_FRAME_SIZE);
180  if(pData->filter_length>0){
181  saf_matrixConv_create(&(pData->hMatrixConv),
182  pData->hostBlockSize_clamped, /*pData->hostBlockSize,*/
183  pData->filters,
184  pData->filter_length,
185  pData->nInputChannels,
186  pData->nOutputChannels,
187  pData->enablePartitionedConv);
188  }
189 
190  /* Resize buffers */
191  pData->inputFrameTD = (float**)realloc2d((void**)pData->inputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
192  pData->outputFrameTD = (float**)realloc2d((void**)pData->outputFrameTD, MAX_NUM_CHANNELS, pData->hostBlockSize_clamped, sizeof(float));
193  memset(FLATTEN2D(pData->inputFrameTD), 0, MAX_NUM_CHANNELS*(pData->hostBlockSize_clamped)*sizeof(float));
194 
195  /* reset FIFO buffers */
196  pData->FIFO_idx = 0;
197  memset(pData->inFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
198  memset(pData->outFIFO, 0, MAX_NUM_CHANNELS*MAX_FRAME_SIZE*sizeof(float));
199 
200  pData->reInitFilters = 0;
201  }
202 }

204 void matrixconv_setFilters
205 (
206  void* const hMCnv,
207  const float** H,
208  int numChannels,
209  int numSamples,
210  int sampleRate
211 )
212 {
213  matrixconv_data *pData = (matrixconv_data*)(hMCnv);
214  int i;
215  saf_assert(numChannels<=MAX_NUM_CHANNELS_FOR_WAV && numChannels > 0 && numSamples > 0, "WAV is limited to 1024 channels");
216 
217  pData->nOutputChannels = SAF_MIN(numChannels, MAX_NUM_CHANNELS);
218  pData->input_wav_length = numSamples;
219  pData->nfilters = (pData->nOutputChannels) * (pData->nInputChannels);
220 
221  /* store the loaded filters */
222  pData->filters = realloc1d(pData->filters, numChannels * numSamples * sizeof(float));
223  for(i=0; i<numChannels; i++)
224  memcpy(&(pData->filters[i*numSamples]), H[i], numSamples * sizeof(float));
225  pData->filter_fs = sampleRate;
226 
227  /* if the number of samples in loaded data is not divisable by the currently specified number of
228  * inputs, then the filter length is set to 0 and no further processing is conducted. */
229  if(pData->input_wav_length % pData->nInputChannels == 0)
230  pData->filter_length = (pData->input_wav_length) / (pData->nInputChannels);
231  else
232  pData->filter_length = 0;
233 
234  pData->reInitFilters = 1;
235 }
